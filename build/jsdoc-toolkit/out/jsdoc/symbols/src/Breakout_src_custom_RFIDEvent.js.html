  6 JSUTILS.namespace('BO.custom.RFIDEvent');
  7 
  8 BO.custom.RFIDEvent = (function() {
  9 	"use strict";
 10 
 11 	var RFIDEvent;
 12 
 13 	// dependencies
 14 	var Event = JSUTILS.Event;
 15 
 16 	/**
 17 	 * @exports RFIDEvent as BO.custom.RFIDEvent
 18 	 * @class An Event object to be dispatched when an RFID tag
 19 	 * is read or when an RFID tag is removed from a reader.
 20 	 * @constructor
 21 	 * @augments JSUTILS.Event
 22 	 * @param {String} type The event type
 23 	 * @param {String} tag The RFID tag value (hexadecimal)
 24 	 */
 25 	RFIDEvent = function(type, tag) {
 26 		this._tag = tag;
 27 		// call the super class
 28 		// 2nd parameter is passed to EventDispatcher constructor
 29 		Event.call(this, type);
 30 
 31 		this.name = "RFIDEvent";
 32 	};
 33 
 34 	/** @constant */
 35 	RFIDEvent.ADD_TAG = "addTag";
 36 	/** @constant */
 37 	RFIDEvent.REMOVE_TAG = "removeTag";
 38 
 39 	RFIDEvent.prototype = JSUTILS.inherit(Event.prototype);
 40 	RFIDEvent.prototype.constructor = RFIDEvent;
 41 
 42 	/**
 43 	 * [read-only] The RFID tag value (hexadecimal string).
 44 	 * @name RFIDEvent#tag
 45 	 * @property
 46 	 * @type String
 47 	 */ 
 48 	RFIDEvent.prototype.__defineGetter__("tag", function() { return this._tag; });
 49 
 50 	return RFIDEvent;
 51 
 52 }());
