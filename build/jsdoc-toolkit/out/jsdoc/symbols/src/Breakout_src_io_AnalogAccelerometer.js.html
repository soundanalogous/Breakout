 14 BO.io.AnalogAccelerometer = (function() {
 15 	"use strict";

 17 	var AnalogAccelerometer;

 19 	// private static constants
 20 	var RAD_TO_DEG = 180 / Math.PI;
 21 
 22 		// dependencies
 23 	var PhysicalInputBase = BO.PhysicalInputBase,
 24 		PinEvent = BO.PinEvent,
 25 		AccelerometerEvent = BO.io.AccelerometerEvent,
 26 		Scaler = BO.filters.Scaler,
 27 		Convolution = BO.filters.Convolution;

 29 	/**
 30 	 * Creates a new Analog Accelerometer object
 31 	 *
 32 	 * @exports AnalogAccelerometer as BO.io.AnalogAccelerometer
 33 	 * @class Creates an interface to an analog accelerometer. Use the
 34 	 * accelerometer to read the acceleration along the x, y, and z axis of an 
 35 	 * object it is attached to. You can also obtain the pitch and roll. This
 36 	 * object should interface with most analog accelerometers.	See
 37 	 * Breakout/examples/sensors/analog_accelerometer.html and 
 38 	 * Breakout/examples/three_js/accelerometer.html for example applications. 
 39 	 * @constructor
 40 	 * @augments BO.PhysicalInputBase	 
 41 	 * @param {IOBoard} board A reference to the IOBoard instance
 42 	 * @param {Pin} xPin A reference to the Pin connected to the x axis of the
 43 	 * accelerometer
 44 	 * @param {Pin} yPin A reference to the Pin connected to the y axis of the
 45 	 * accelerometer
 46 	 * @param {Pin} zPin A reference to the Pin connected to the z axis of the
 47 	 * accelerometer
 48 	 * @param {Number} dynamicRange The range of the acceleromter in Gs
 49 	 * (typically 2 or 3 for an 
 50 	 * analog accelerometer). See the datasheet for the acceleromter to get
 51 	 * the exact value.
 52 	 * @param {Boolean} enableSmoothing True to enable smoothing, false to
 53 	 * disable. Default is false.
 54 	 */
 55 	AnalogAccelerometer = function(board, xPin, yPin, zPin, dynamicRange, enableSmoothing) {
 56 
 57 		// Call the super class
 58 		PhysicalInputBase.call(this);
 59 
 60 		this.name = "AnalogAccelerometer";
 61 
 62 		// Enable the analog pins
 63 		board.enableAnalogPin(xPin.analogNumber);
 64 		board.enableAnalogPin(yPin.analogNumber);
 65 		board.enableAnalogPin(zPin.analogNumber);		
 66 
 67 		this._enableSmoothing = enableSmoothing || false;
 68 		this._xPin = xPin || null;
 69 		this._yPin = yPin || null;
 70 		this._zPin = zPin || null;
 71 
 72 		// Common accelerometer interface values:
 73 		this._dynamicRange = dynamicRange || 1;
 74 		this._x = 0;
 75 		this._y = 0;
 76 		this._z = 0;
 77 
 78 		if (this._xPin !== null) {
 79 			this._xPin.addEventListener(PinEvent.CHANGE, this.xAxisChanged.bind(this));
 80 		}
 81 
 82 		if (this._yPin !== null) {
 83 			this._yPin.addEventListener(PinEvent.CHANGE, this.yAxisChanged.bind(this));
 84 		}
 85 		
 86 		if (this._zPin !== null) {
 87 			this._zPin.addEventListener(PinEvent.CHANGE, this.zAxisChanged.bind(this));
 88 		}
 89 		
 90 	};
 91 
 92 	AnalogAccelerometer.prototype = JSUTILS.inherit(PhysicalInputBase.prototype);
 93 	AnalogAccelerometer.prototype.constructor = AnalogAccelerometer;

118 	 * [read-only] The y axis of the accelerometer in units 
119 	 * of gravity (9.8 m/sec2).
120 	 * 
121 	 * @name AnalogAccelerometer#y

123 	 * @type Number
124 	 */

128 	 * [read-only] The z axis of the accelerometer in units 
129 	 * of gravity (9.8 m/sec2).

131 	 * @name AnalogAccelerometer#z

137 	/**
138 	 * [read-only] The pitch value in degrees.
139 	 * 
140 	 * @name AnalogAccelerometer#pitch
141 	 * @property
142 	 * @type Number
143 	 */ 
144 	AnalogAccelerometer.prototype.__defineGetter__("pitch", function() { 
145 		// -180 to 180
146 		//return Math.atan2(this._x, this._z) * RAD_TO_DEG;
147 		// -90 to 90
148 		return Math.atan2(this._x, Math.sqrt(this._y * this._y + this._z * this._z)) * RAD_TO_DEG;
149 	});
150 	
151 	/**
152 	 * [read-only] The roll value in degrees.
153 	 * 
154 	 * @name AnalogAccelerometer#roll
155 	 * @property
156 	 * @type Number
157 	 */ 
158 	AnalogAccelerometer.prototype.__defineGetter__("roll", function() { 
159 		// -180 to 180
160 		//return Math.atan2(this._y, this._z) * RAD_TO_DEG;
161 		// -90 to 90
162 		return Math.atan2(this._y, Math.sqrt(this._x * this._x + this._z * this._z)) * RAD_TO_DEG;
163 	});

166 
167 	AnalogAccelerometer.prototype.__defineGetter__("xPin", function() { 
168 		return this._xPin;
169 	});
170 
171 	AnalogAccelerometer.prototype.__defineGetter__("yPin", function() { 
172 		return this._yPin;
173 	});

175 	AnalogAccelerometer.prototype.__defineGetter__("zPin", function() { 
176 		return this._zPin;
177 	});

179 	/**
180 	 * Scale the range for the specified axis (from 0 to 1) to (minimum to 
181 	 * maximum).
182 	 * 
183 	 * @param axis the axis to set new range (AnalogAccelerometer.X_AXIS, 
184 	 * AnalogAccelerometer.Y_AXIS or AnalogAccelerometer.Z_AXIS).
185 	 * @param {Number} minimum The new minimum value
186 	 * @param {Number} maximum The new maximum value
187 	 */
188 	AnalogAccelerometer.prototype.setRangeFor = function(axis, minimum, maximum) {
189 		var range = this._dynamicRange;
190 
191 		if (axis === AnalogAccelerometer.X_AXIS) {
192 			if (this._xPin !== null) {
193 				this._xPin.filters = [new Scaler(minimum, maximum, -range, range, Scaler.LINEAR)];
194 				if (this._enableSmoothing) {
195 					this._xPin.addFilter(new Convolution(Convolution.MOVING_AVERAGE));
196 				}
197 			}
198 		} else if (axis === AnalogAccelerometer.Y_AXIS) {
199 			if (this._yPin !== null) {
200 				this._yPin.filters = [new Scaler(minimum, maximum, -range, range, Scaler.LINEAR)];
201 				if (this._enableSmoothing) {
202 					this._yPin.addFilter(new Convolution(Convolution.MOVING_AVERAGE));
203 				}
204 			}
205 		} else if (axis === AnalogAccelerometer.Z_AXIS) {
206 			if (this._zPin !== null) {
207 				this._zPin.filters = [new Scaler(minimum, maximum, -range, range, Scaler.LINEAR)];
208 				if (this._enableSmoothing) {
209 					this._zPin.addFilter(new Convolution(Convolution.MOVING_AVERAGE));
210 				}
211 			}
212 		}
213 	};
214 
215 	// Calibration:
216 	// For each axis, calculate variance from -1 and 1
217 	// assume zero g = supply voltage/2 and sensitivity is supply voltage / 10 per g
218 	// at -1 g, voltage should be (supply voltage/2) - (supply voltage / 10)
219 	// at 1 g, voltage should be (supply voltage/2) + (supply voltage / 10)
220 
221 	/**
222 	 * Use this method to get the minimum and maximum range values for an axis.
223 	 * Create a new object to store the return value and then pass obj.min
224 	 * and obj.max along with the respective axis identifier to the setRangeFor
225 	 * method.
226 	 * 
227 	 * @param {Number} minVoltage The minimum value reported on the axis
228 	 * @param {Number} maxVoltage The maximum value reported on the axis
229 	 * @param {Number} supplyVoltage The supply voltage of the Acceleromter
230 	 * (enter as 3.3, 3.0, 5.0, etc).
231 	 * @return {Object} An object containing the min and max range values to be
232 	 * passed to the setRangeFor method.
233 	 */
234 	AnalogAccelerometer.prototype.getCalibratedRange = function(minVoltage, maxVoltage, supplyVoltage) {
235 		var range = {min:0, max:0};
236 		
237 		var mVPerG = (maxVoltage - minVoltage) / 2;
238 		
239 		// Find zero G (average of min and max)
240 		var zeroG = (minVoltage + maxVoltage) / 2;
241 		
242 		range.min = (zeroG - (mVPerG * this._dynamicRange))/supplyVoltage;
243 		range.max = (zeroG + (mVPerG * this._dynamicRange))/supplyVoltage;
244 		
245 		return range;
246 	};

252 		this._x = event.target.value;

254 	};
255

257 	 * @private
258 	 */

260 		this._y = event.target.value;
261 		this.dispatchEvent(new AccelerometerEvent(AccelerometerEvent.UPDATE));
262 	};
263 	
264 	/**
265 	 * @private
266 	 */
267 	AnalogAccelerometer.prototype.zAxisChanged = function(event) {
268 		this._z = event.target.value;
269 		this.dispatchEvent(new AccelerometerEvent(AccelerometerEvent.UPDATE));
270 	};		
271

274 	/** @constant */
275 	AnalogAccelerometer.Y_AXIS = 1;
276 	/** @constant */
277 	AnalogAccelerometer.Z_AXIS = 2;
