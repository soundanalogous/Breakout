  5 /**
  6  * An as3-like EventDispatcher class.
  7  *
  8  * @constructor
  9  * @param {Class} target The instance of the class that implements EventDispatcher
 10  */
 11 function EventDispatcher(target) {
 12 	"use strict";
 13 	
 14 	var _target = target || null;
 15 	var _eventListeners = {};
 16 	
 17 	/**
 18 	 * @param {String} type The event type
 19 	 * @param {Function} listener The function to be called when the event is fired
 20 	 */
 21 	this.addEventListener = function (type, listener) {
 22 		if(!_eventListeners[type]) {
 23 			_eventListeners[type] = [];
 24 		}
 25 		_eventListeners[type].push(listener);
 26 	}
 27 	
 28 	/**
 29 	 * @param {String} type The event type
 30 	 * @param {Function} listener The function to be called when the event is fired
 31 	 */
 32 	this.removeEventListener = function(type, listener) {
 33 		for (var i=0, len = _eventListeners[type].length; i<len; i++) {
 34 			if (_eventListeners[type][i] == listener) {
 35 				_eventListeners[type].splice(i, 1);
 36 			}
 37 		}
 38 	}
 39 	
 40 	/**
 41 	 * @param {String} type The event type
 42 	 * return {boolean} True is listener exists for this type, false if not.
 43 	 */
 44 	this.hasEventListener = function(type) {
 45 		// to do: implement this method
 46 		if (_eventListeners[type]) {
 47 			return true;
 48 		} else {
 49 			return false;
 50 		}	
 51 	}
 52 	
 53 	/**
 54 	 * @param {Event} type The Event object
 55 	 * return {boolean} True if dispatch is successful, false if not.
 56 	 */	
 57 	this.dispatchEvent = function(event) {
 58 		event.target = _target;
 59 		var isSuccess = false;
 60 					
 61 		if (this.hasEventListener(event.type)) {
 62 			for (var j=0, len=_eventListeners[event.type].length; j<len; j++) {
 63 				try {
 64 					_eventListeners[event.type][j].call(this, event);
 65 					isSuccess = true;
 66 				} catch(e) {
 67 					// to do: handle error
 68 				}
 69 			}
 70 		}
 71 		return isSuccess;	
 72 	}
 73 }
 74 
 75 /** 
 76  * Event 'base class' (but it can also be instantiated directly)
 77  *
 78  * @constructor
 79  * @param {String} type event type
 80  */
 81 function Event(type) {
 82 	this.type = type;
 83 	this.target = null;
 84 }
