 12  JSUTILS.namespace('BO.generators.Oscillator');
 13 
 14  BO.generators.Oscillator = (function() {
 15  	"use strict";
 16 
 17  	var Oscillator;
 18 
 19  	// dependencies
 20  	var GeneratorBase = BO.generators.GeneratorBase,
 21  		GeneratorEvent = BO.generators.GeneratorEvent,
 22  		Timer = JSUTILS.Timer,
 23  		TimerEvent = JSUTILS.TimerEvent;
 24 
 25  	/**
 26 	 * Osc outputs a waveform on the associated PWM pin. For example, this can be used to blink or fade
 27 	 * an LED on or off.
 28 	 *
 29  	 * @exports Oscillator as BO.generators.Oscillator
 30  	 * @constructor
 31  	 * @augments BO.generators.GeneratorBase
 32 	 * @param {Number} wave waveform
 33 	 * @param {Number} freq frequency
 34 	 * @param {Number} amplitude amplitude
 35 	 * @param {Number} offset offset
 36 	 * @param {Number} phase phase
 37 	 * @param {Number} times The repeat count from 0 to infinite.
 38  	 */
 39  	Oscillator = function(wave, freq, amplitude, offset, phase, times) {
 40 
 41  		// call super class
 42  		GeneratorBase.call(this);
 43 
 44  		this.name = "Oscillator";
 45 
 46  		this._wave = wave || Oscillator.SIN;
 47  		this._freq = freq || 1;
 48  		this._amplitude = amplitude || 1;
 49  		this._offset = offset || 0;
 50  		this._phase = phase || 0;
 51  		this._times = times || 0;
 52 
 53  		if (freq ===0) throw new Error("Frequency should be larger than 0");
 54 
 55  		this._time;
 56  		this._startTime;
 57  		this._lastVal;
 58  		// need to do this in order to remove the event listener
 59  		this._autoUpdateCallback = this.autoUpdate.bind(this);
 60 
 61  		this._timer = new Timer(33);
 62  		this._timer.start();
 63 
 64  		this.reset();
 65  	};
 66 
 67  	Oscillator.prototype = JSUTILS.inherit(GeneratorBase.prototype);
 68  	Oscillator.prototype.constructor = Oscillator;
 69 
 70 	/**
 71 	 * The service interval in milliseconds. Default is 33ms.
 72 	 *
 73 	 * @name Oscillator#serviceInterval
 74 	 * @property
 75 	 * @type Number
 76 	 */ 
 77  	Oscillator.prototype.__defineSetter__("serviceInterval", function(interval) {
 78  		this._timer.delay = interval; 		
 79  	});
 80  	Oscillator.prototype.__defineGetter__("serviceInterval", function() {
 81  		return this._timer.delay;
 82  	});
 83 
 84  	/**
 85  	 * Starts the oscillator
 86  	 */
 87  	Oscillator.prototype.start = function() {
 88  		this.stop();
 89  		this._timer.addEventListener(TimerEvent.TIMER, this._autoUpdateCallback);
 90 
 91  		var date = new Date();
 92  		this._startTime = date.getTime();
 93  		this.autoUpdate(null);
 94  	};
 95 
 96  	/**
 97  	 * Stops the oscillator.
 98  	 */
 99  	Oscillator.prototype.stop = function() {
100  		if (this._timer.hasEventListener(TimerEvent.TIMER)) {
101  			this._timer.removeEventListener(TimerEvent.TIMER, this._autoUpdateCallback);
102  		}
103  	};
104 
105  	/**
106  	 * Resets the oscillator.
107  	 */
108  	Oscillator.prototype.reset = function() {
109  		this._time = 0;
110  		this._lastVal = 0.999;
111  	};
112 
113  	/**
114  	 * By default the interval is 33 milliseconds. The Osc is updated every 33ms.
115  	 * @param {Number} interval The update interval in milliseconds.
116  	 */
117  	Oscillator.prototype.update = function(interval) {
118  		interval = interval || -1;
119  		if (interval < 0) this._time += this._timer.delay;
120  		else this._time += interval;
121  		this.computeValue();
122  	};
123 
124  	/**
125  	 * @private
126  	 */
127  	Oscillator.prototype.autoUpdate = function(event) {
128  		var date = new Date();
129  		this._time = date.getTime() - this._startTime;
130  		this.computeValue();
131  	};
132 
133  	/**
134  	 * @private
135  	 */
136  	Oscillator.prototype.computeValue = function() {
137  		var sec = this._time / 1000;
138 
139  		if (this._times !== 0 && this._freq * sec >= this._times) {
140  			this.stop();
141  			sec = this._times / this._freq;
142  			if (this._wave !== Oscillator.LINEAR) {
143  				this._value = this._offset;
144  			} else {
145  				this._value = this._amplitude * this._wave(1, 0) + this._offset;
146  			}
147  		} else {
148  			var val = this._freq * (sec + this._phase);
149  			this._value = this._amplitude * this._wave(val, this._lastVal) + this._offset;
150  			this._lastVal = val;
151  		}
152  		this.dispatchEvent(new GeneratorEvent(GeneratorEvent.UPDATE));
153  	};
154 
155  	// Static methods
156 
157  	/**
158  	 * sine wave
159  	 * @static
160  	 */
161  	Oscillator.SIN = function(val, lastVal) {
162  		return 0.5 * (1 + Math.sin(2 * Math.PI * (val - 0.25)));
163  	};
164 
165  	/**
166  	 * square wave
167  	 * @static
168  	 */
169  	Oscillator.SQUARE = function(val, lastVal) {
170  		return (val % 1 <= 0.5) ? 1 : 0;
171  	};
172  	
173  	/**
174  	 * triangle wave
175  	 * @static
176  	 */
177  	Oscillator.TRIANGLE = function(val, lastVal) {
178  		val %= 1;
179  		return (val <= 0.5) ? (2 * val) : (2 - 2 * val);
180  	};
181  	
182  	/**
183  	 * saw wave
184  	 * @static
185  	 */
186  	Oscillator.SAW = function(val, lastVal) {
187  		val %= 1;
188  		if (val <= 0.5) return val + 0.5;
189  		else return val - 0.5;
190  	};
191  	
192  	/**
193  	 * impulse
194  	 * @static
195  	 */
196  	Oscillator.IMPULSE = function(val, lastVal) {
197  		return ((val % 1) < (lastVal % 1)) ? 1 : 0;
198  	};
199  	
200  	/**
201  	 * linear
202  	 * @static
203  	 */
204  	Oscillator.LINEAR = function(val, lastVal) {
205  		return (val < 1) ? val : 1;
206  	};
207  	
208 	// document events
209 
210 	/**
211 	 * The update event is dispatched at the rate specified 
212 	 * by the serviceInterval parameter (default = 33ms).
213 	 * @name Oscillator#update
214 	 * @type BO.generators.GeneratorEvent.UPDATE
215 	 * @event
216 	 * @param {BO.generators.Oscillator} target A reference to the Oscillator object.
217 	 */	 		 	
218 
219  	return Oscillator;
220 
221 
222  }());
