 12  JSUTILS.namespace('BO.generators.Oscillator');
 13 
 14  BO.generators.Oscillator = (function() {
 15  	"use strict";
 16 
 17  	var Oscillator;
 18 
 19  	// dependencies
 20  	var GeneratorBase = BO.generators.GeneratorBase,
 21  		GeneratorEvent = BO.generators.GeneratorEvent,
 22  		Timer = JSUTILS.Timer,
 23  		TimerEvent = JSUTILS.TimerEvent;
 24 
 25  	/**
 26 	 * Osc outputs a waveform on the associated PWM pin. For example, this can be used to blink or fade
 27 	 * an LED on or off.
 28 	 *
 29  	 * @exports Oscillator as BO.generators.Oscillator
 30  	 * @class The Oscillator object can be attached to a Pin or LED object to output
 31  	 * a waveform. This is useful for blinking an LED or fading it on and off. In
 32  	 * most cases (unless you are simply using it to blink and LED on or off), 
 33  	 * the Oscillator should be attached to a Pin or LED object associated with
 34  	 * a PWM pin on the I/O board. See Breakout/examples/generators/oscillator.html
 35  	 * for an example application.
 36  	 * @constructor
 37  	 * @augments BO.generators.GeneratorBase
 38 	 * @param {Number} wave waveform
 39 	 * @param {Number} freq frequency
 40 	 * @param {Number} amplitude amplitude
 41 	 * @param {Number} offset offset
 42 	 * @param {Number} phase phase
 43 	 * @param {Number} times The repeat count from 0 to infinite.
 44  	 */
 45  	Oscillator = function(wave, freq, amplitude, offset, phase, times) {
 46 
 47  		// call super class
 48  		GeneratorBase.call(this);
 49 
 50  		this.name = "Oscillator";
 51 
 52  		this._wave = wave || Oscillator.SIN;
 53  		this._freq = freq || 1;
 54  		this._amplitude = amplitude || 1;
 55  		this._offset = offset || 0;
 56  		this._phase = phase || 0;
 57  		this._times = times || 0;
 58 
 59  		if (freq ===0) throw new Error("Frequency should be larger than 0");
 60 
 61  		this._time;
 62  		this._startTime;
 63  		this._lastVal;
 64  		// need to do this in order to remove the event listener
 65  		this._autoUpdateCallback = this.autoUpdate.bind(this);
 66 
 67  		this._timer = new Timer(33);
 68  		this._timer.start();
 69 
 70  		this.reset();
 71  	};
 72 
 73  	Oscillator.prototype = JSUTILS.inherit(GeneratorBase.prototype);
 74  	Oscillator.prototype.constructor = Oscillator;
 75 
 76 	/**
 77 	 * The service interval in milliseconds. Default is 33ms.
 78 	 *
 79 	 * @name Oscillator#serviceInterval
 80 	 * @property
 81 	 * @type Number
 82 	 */ 
 83  	Oscillator.prototype.__defineSetter__("serviceInterval", function(interval) {
 84  		this._timer.delay = interval; 		
 85  	});
 86  	Oscillator.prototype.__defineGetter__("serviceInterval", function() {
 87  		return this._timer.delay;
 88  	});
 89 
 90  	/**
 91  	 * Starts the oscillator
 92  	 */
 93  	Oscillator.prototype.start = function() {
 94  		this.stop();
 95  		this._timer.addEventListener(TimerEvent.TIMER, this._autoUpdateCallback);
 96 
 97  		var date = new Date();
 98  		this._startTime = date.getTime();
 99  		this.autoUpdate(null);
100  	};
101 
102  	/**
103  	 * Stops the oscillator.
104  	 */
105  	Oscillator.prototype.stop = function() {
106  		if (this._timer.hasEventListener(TimerEvent.TIMER)) {
107  			this._timer.removeEventListener(TimerEvent.TIMER, this._autoUpdateCallback);
108  		}
109  	};
110 
111  	/**
112  	 * Resets the oscillator.
113  	 */
114  	Oscillator.prototype.reset = function() {
115  		this._time = 0;
116  		this._lastVal = 0.999;
117  	};
118 
119  	/**
120  	 * By default the interval is 33 milliseconds. The Osc is updated every 33ms.
121  	 * @param {Number} interval The update interval in milliseconds.
122  	 */
123  	Oscillator.prototype.update = function(interval) {
124  		interval = interval || -1;
125  		if (interval < 0) this._time += this._timer.delay;
126  		else this._time += interval;
127  		this.computeValue();
128  	};
129 
130  	/**
131  	 * @private
132  	 */
133  	Oscillator.prototype.autoUpdate = function(event) {
134  		var date = new Date();
135  		this._time = date.getTime() - this._startTime;
136  		this.computeValue();
137  	};
138 
139  	/**
140  	 * @private
141  	 */
142  	Oscillator.prototype.computeValue = function() {
143  		var sec = this._time / 1000;
144 
145  		if (this._times !== 0 && this._freq * sec >= this._times) {
146  			this.stop();
147  			sec = this._times / this._freq;
148  			if (this._wave !== Oscillator.LINEAR) {
149  				this._value = this._offset;
150  			} else {
151  				this._value = this._amplitude * this._wave(1, 0) + this._offset;
152  			}
153  		} else {
154  			var val = this._freq * (sec + this._phase);
155  			this._value = this._amplitude * this._wave(val, this._lastVal) + this._offset;
156  			this._lastVal = val;
157  		}
158  		this.dispatchEvent(new GeneratorEvent(GeneratorEvent.UPDATE));
159  	};
160 
161  	// Static methods
162 
163  	/**
164  	 * sine wave
165  	 * @static
166  	 */
167  	Oscillator.SIN = function(val, lastVal) {
168  		return 0.5 * (1 + Math.sin(2 * Math.PI * (val - 0.25)));
169  	};
170 
171  	/**
172  	 * square wave
173  	 * @static
174  	 */
175  	Oscillator.SQUARE = function(val, lastVal) {
176  		return (val % 1 <= 0.5) ? 1 : 0;
177  	};
178  	
179  	/**
180  	 * triangle wave
181  	 * @static
182  	 */
183  	Oscillator.TRIANGLE = function(val, lastVal) {
184  		val %= 1;
185  		return (val <= 0.5) ? (2 * val) : (2 - 2 * val);
186  	};
187  	
188  	/**
189  	 * saw wave
190  	 * @static
191  	 */
192  	Oscillator.SAW = function(val, lastVal) {
193  		val %= 1;
194  		if (val <= 0.5) return val + 0.5;
195  		else return val - 0.5;
196  	};
197  	
198  	/**
199  	 * impulse
200  	 * @static
201  	 */
202  	Oscillator.IMPULSE = function(val, lastVal) {
203  		return ((val % 1) < (lastVal % 1)) ? 1 : 0;
204  	};
205  	
206  	/**
207  	 * linear
208  	 * @static
209  	 */
210  	Oscillator.LINEAR = function(val, lastVal) {
211  		return (val < 1) ? val : 1;
212  	};
213  	
214 	// document events
215 
216 	/**
217 	 * The update event is dispatched at the rate specified 
218 	 * by the serviceInterval parameter (default = 33ms).
219 	 * @name Oscillator#update
220 	 * @type BO.generators.GeneratorEvent.UPDATE
221 	 * @event
222 	 * @param {BO.generators.Oscillator} target A reference to the Oscillator object.
223 	 */	 		 	
224 
225  	return Oscillator;
226 
227 
228  }());
