  6 JSUTILS.namespace('BO.I2CBase');
  7 
  8 BO.I2CBase = (function() {
  9 	"use strict";
 10 
 11  	var I2CBase;
 12 
 13  	// dependencies
 14  	var Pin = BO.Pin,
 15  		EventDispatcher = JSUTILS.EventDispatcher,
 16  		IOBoardEvent = BO.IOBoardEvent;
 17 
 18 	/**
 19 	 * Creates a new I2CBase base class
 20 	 *
 21 	 * @exports I2CBase as BO.I2CBase
 22 	 * @class A base class for I2C objects. Extend this class
 23 	 * when creating an interface for a new I2C device. I2CBase should
 24 	 * not be instantiated directly.
 25 	 * @constructor
 26 	 * @param {IOBoard} board A reference to the IOBoard instance
 27 	 * @param {Number} address The I2C address of the device
 28 	 * @param {Number} delayUS The number of microseconds ...
 29 	 */
 30 	I2CBase = function(board, address, delayUS) {
 31 
 32 		if (board == undefined) return;
 33 
 34 		this.name = "I2CBase";
 35 		/** @protected*/
 36 		this.board = board;
 37 
 38 		var _delay = delayUS || 0,
 39 			_delayInMicrosecondsLSB = _delay & 0xFF,
 40 			_delayInMicrosecondsMSB = (_delay >> 8) & 0xFF;
 41 
 42 		/** @protected */
 43 		this._address = address;
 44 		this._evtDispatcher = new EventDispatcher(this);
 45 		
 46 		// if the pins are not set as I2C, set them now
 47 		var i2cPins = board.getI2cPins();
 48 		if (i2cPins.length == 2) {
 49 			if (board.getPin(i2cPins[0]).getType() != Pin.I2C) {
 50 				board.getPin(i2cPins[0]).setType(Pin.I2C);
 51 				board.getPin(i2cPins[1]).setType(Pin.I2C);
 52 			}
 53 		} else {
 54 			// to do: proper error handling
 55 			console.log("Error, this board does not support i2c");
 56 			return;
 57 		}
 58 
 59 		board.addEventListener(IOBoardEvent.SYSEX_MESSAGE, this.onSysExMessage.bind(this));
 60 		
 61 		// call this for each board in case delay is set
 62 		board.sendSysex(I2CBase.I2C_CONFIG, [_delayInMicrosecondsLSB, _delayInMicrosecondsMSB]);
 63 			
 64 	};
 65 
 66 
 67 	I2CBase.prototype = {
 68 
 69 		/**
 70 		 * [read-only] The address of the i2c device.
 71 		 * @name I2CBase#address
 72 		 * @property
 73 		 * @type Number
 74 		 */			 
 75 		get address() {
 76 			return this._address;
 77 		},	
 78 
 79 		// private methods:
 80 		/**
 81 		 * @private
 82 		 */
 83 		onSysExMessage: function(event) {
 84 			var message = event.message;
 85 			var addr = this.board.getValueFromTwo7bitBytes(message[1], message[2]);
 86 			var data = [];
 87 
 88 			if (message[0] != I2CBase.I2C_REPLY) {
 89 				return;
 90 			} else {
 91 				//console.log(this);
 92 				//console.log("addr = " + this._address);
 93 				// to do: make sure i2c address in message matches the i2c address of the subclass
 94 				// return if no match;
 95 				if (addr != this._address) return;
 96 
 97 				for (var i=3, len=message.length; i<len; i+=2) {
 98 					data.push(this.board.getValueFromTwo7bitBytes(message[i], message[i+1]));
 99 				}
100 				this.handleI2C(data);
101 			}
102 
103 		},
104 		
105 		// public methods:
106 		
107 		/**
108 		 * Send an i2c request command to the board
109 		 * @protected
110 		 * @param {Number} command
111 		 * @param {Number[]} data
112 		 */
113 		sendI2CRequest: function(data) {
114 
115 			// to do: support 10-bit i2c address
116 			var tempData = [];
117 			var address = data[1];
118 			var readWriteMode = data[0];
119 			
120 			tempData[0] = address;
121 			tempData[1] = readWriteMode << 3;
122 			
123 			for (var i=2, len=data.length; i<len; i++) {
124 				tempData.push(data[i] & 0x007F);
125 				tempData.push((data[i] >> 7) & 0x007F);				
126 			}
127 			
128 			this.board.sendSysex(I2CBase.I2C_REQUEST, tempData);
129 			
130 		},
131 	
132 		/**
133 		 * @protected
134 		 * @inheritDoc
135 		 */
136 		update: function() {
137 			// To be implemented in sublasses	
138 		},
139 		
140 		/**
141 		 * @protected
142 		 * @inheritDoc
143 		 */
144 		handleI2C: function(data) {
145 			// To be implemented in sublasses
146 			// data should be: slave address, register, data0, data1...
147 		},
148 				
149 		/* implement EventDispatcher */
150 		
151 		/**
152 		 * @param {String} type The event type
153 		 * @param {Function} listener The function to be called when the event is fired
154 		 */
155 		addEventListener: function(type, listener) {
156 			this._evtDispatcher.addEventListener(type, listener);
157 		},
158 		
159 		/**
160 		 * @param {String} type The event type
161 		 * @param {Function} listener The function to be called when the event is fired
162 		 */
163 		removeEventListener: function(type, listener) {
164 			this._evtDispatcher.removeEventListener(type, listener);
165 		},
166 		
167 		/**
168 		 * @param {String} type The event type
169 		 * return {boolean} True is listener exists for this type, false if not.
170 		 */
171 		hasEventListener: function(type) {
172 			return this._evtDispatcher.hasEventListener(type);
173 		},
174 		
175 		/**
176 		 * @param {Event} type The Event object
177 		 * @param {Object} optionalParams Optional parameters to assign to the event object.
178 		 * return {boolean} True if dispatch is successful, false if not.
179 		 */		
180 		dispatchEvent: function(event, optionalParams) {
181 			return this._evtDispatcher.dispatchEvent(event, optionalParams);
182 		}			
183 	};
184 	
185 
186 	/** @constant */
187 	I2CBase.I2C_REQUEST = 0x76;
188 	/** @constant */
189 	I2CBase.I2C_REPLY = 0x77;
190 	/** @constant */
191 	I2CBase.I2C_CONFIG = 0x78;
192 
193 	/** @constant */
194 	I2CBase.WRITE = 0;
195 	/** @constant */
196 	I2CBase.READ = 1;
197 	/** @constant */
198 	I2CBase.READ_CONTINUOUS = 2;
199 	/** @constant */
200 	I2CBase.STOP_READING = 3;
201 
202 	return I2CBase;
203 
204 }());
