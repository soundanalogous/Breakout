 12 JSUTILS.namespace('BREAKOUT.io.BlinkM');
 13 
 14 BREAKOUT.io.BlinkM = (function() {
 15 
 16 	var BlinkM;
 17 
 18 	// dependencies
 19 	var I2CBase = BREAKOUT.I2CBase;
 20 
 21 	/**
 22 	 * Creates and BlinkM object
 23 	 *
 24 	 * @exports BlinkM as BREAKOUT.io.BlinkM
 25 	 * @constructor
 26 	 * @augments BREAKOUT.I2CBase 
 27 	 * @param {IOBoard} board The IOBoard instance
 28 	 * @param {Number} address The i2c address of the BlinkM module
 29 	 */
 30 	BlinkM = function(board, address) {
 31 
 32 		address = address || 0x09;	// default i2c address for BlinkM
 33 
 34 		this.name = "BlinkM"; // for testing
 35 		
 36 		// call super class
 37 		I2CBase.call(this, board, address);
 38 		
 39 
 40 	};
 41 
 42 	BlinkM.prototype = JSUTILS.inherit(I2CBase.prototype);
 43 	BlinkM.prototype.constructor = BlinkM;
 44 
 45 
 46 	/**
 47 	 * Sets the BlinkM to the specified RGB color immediately
 48 	 * @param {Number{}} color An array containing the RGB values. 
 49 	 * color[0] = R, color[1] = G, color[2] = B
 50 	 */
 51 	BlinkM.prototype.goToRGBColorNow = function(color) {
 52 		this.sendI2CRequest([I2CBase.WRITE, this.address, 0x6E, color[0], color[1], color[2]]);
 53 	};
 54 	
 55 	/**
 56 	 * Fades to the specified RGB color in the specified time duration. 
 57 	 * The fade speed range is from 1 to 255, where 1 is the slowest time and 255 is the fastest.
 58 	 *
 59 	 * @param {Number[]} color An array containing the RGB values.
 60 	 * color[0] = R, color[1] = G, color[2] = B
 61 	 * @param {Number} speed The fade speed. Default value is 15.
 62 	 */
 63 	BlinkM.prototype.fadeToRGBColor = function(color, speed) {
 64 		var speed = speed || -1;
 65 		if (speed >= 0) {
 66 			this.sendI2CRequest([I2CBase.WRITE, this.address, 0x66, speed]);
 67 		}
 68 		this.sendI2CRequest([I2CBase.WRITE, this.address, 0x63, color[0], color[1], color[2]]);
 69 	};
 70 
 71 	/**
 72 	 * Fade to a random RGB color.
 73 	 * The fade speed range is from 1 to 255, where 1 is the slowest time and 255 is the fastest.
 74 	 *
 75 	 * @param {Number[]} colorRange An array containing a range for each color value.
 76 	 * colorRange[0] = range for Red (0-255), colorRange[1] = range for Green, etc.
 77 	 * @param {Number} speed The fade speed. Default value is 15.
 78 	 */
 79 	BlinkM.prototype.fadeToRandomRGBColor = function (colorRange, speed) {
 80 		var speed = speed || -1;
 81 		if (speed >= 0) {
 82 			this.sendI2CRequest([I2CBase.WRITE, this.address, 0x66, speed]);
 83 		}
 84 		this.sendI2CRequest([I2CBase.WRITE, this.address, 0x43, colorRange[0], colorRange[1], colorRange[2]]);
 85 	};
 86 
 87 	/**
 88 	 * Fades to the specified HSB color in the specified time duration. 
 89 	 * The fade speed range is from 1 to 255, where 1 is the slowest time and 255 is the fastest.
 90 	 *
 91 	 * @param {Number[]} color An array containing the HSB values.
 92 	 * color[0] = H, color[1] = S, color[2] = B
 93 	 * @param {Number} speed The fade speed. Default value is 15.
 94 	 */
 95 	BlinkM.prototype.fadeToHSBColor = function(color, speed) {
 96 		var speed = speed || -1;
 97 		if (speed >= 0) {
 98 			this.sendI2CRequest([I2CBase.WRITE, this.address, 0x66, speed]);
 99 		}
100 		this.sendI2CRequest([I2CBase.WRITE, this.address, 0x68, color[0], color[1], color[2]]);
101 	};
102 	
103 	/**
104 	 * Fade to a random HSB color.
105 	 * The fade speed range is from 1 to 255, where 1 is the slowest time and 255 is the fastest.
106 	 *
107 	 * @param {Number[]} colorRange An array containing a range for each color value.
108 	 * colorRange[0] = range for Hue (0-255), colorRange[1] = range for Saturation, etc.
109 	 * @param {Number} speed The fade speed. Default value is 15.
110 	 */	
111 	BlinkM.prototype.fadeToRandomHSBColor = function(colorRange, speed) {
112 		var speed = speed || -1;
113 		if (speed >= 0) {
114 			this.sendI2CRequest([I2CBase.WRITE, this.address, 0x66, speed]);
115 		}
116 		this.sendI2CRequest([I2CBase.WRITE, this.address, 0x48, colorRange[0], colorRange[1], colorRange[2]]);
117 	};
118 
119 	/**
120 	 * Set the rate at which color fading happens. The range is from 1 to 255, where 1 is the
121 	 * slowest and 255 is the fastest (immediate).
122 	 *
123 	 * @param {Number} speed
124 	 */
125 	BlinkM.prototype.setFadeSpeed = function(speed) {
126 		this.sendI2CRequest([I2CBase.WRITE, this.address, 0x66, speed]);
127 	};
128 	
129 	/**
130 	 * Play a predefined light script. See the BlinkM datasheet page 20 for a list and 
131 	 * description of the predefined scripts.
132 	 *
133 	 * @param {Number} scriptId The id of the light script (from 0 to 18).
134 	 * @param {Number} theNumberOfRepeats The number of times the script should repeat.
135 	 * @param {Number} lineNumber The line number to begin the script from.
136 	 */
137 	BlinkM.prototype.playLightScript = function(scriptId, theNumberOfRepeats, lineNumber) {
138 		var theNumberOfRepeats = theNumberOfRepeats || 1;
139 		this.sendI2CRequest([I2CBase.WRITE, this.address, 0x70, scriptId, theNumberOfRepeats, lineNumber]);
140 	};
141 
142 	/**
143 	 * Stop the currently playing predefined light script.
144 	 */
145 	BlinkM.prototype.stopScript = function() {
146 		//self.sendI2CRequest([I2CBase.WRITE, this.address, 'o'.charCodeAt(0)]);
147 		this.sendI2CRequest([I2CBase.WRITE, this.address, 0x6F]);
148 	};
149 
150 	/**
151 	 * @private
152 	 */
153 	BlinkM.prototype.handleI2C = function(data) {
154 		// TODO: implement if needed
155 		console.log("BlinkM: " + data);
156 	};
157 
158 	return BlinkM;
159 
160 }());
