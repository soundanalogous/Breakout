153 		/**
154 		 * @private
155 		 */
156 		function processData(inputData) {
157 			inputData *= 1;	// force inputData to integer (is there a better way to do this?)
158 			var command;
159 			
160 			// we have command data
161 			if (_waitForData > 0 && inputData < 128) {
162 				_waitForData--;
163 				// collect the data
164 				_storedInputData[_waitForData]=inputData;
165 				// we have all data executeMultiByteCommand
166 				if (_waitForData == 0) {
167 					switch (_executeMultiByteCommand) {
168 						case DIGITAL_MESSAGE:
169 							processDigitalPortBytes(_multiByteChannel, _storedInputData[1], _storedInputData[0]); //(LSB, MSB)
170 							break;
171 						case REPORT_VERSION:
172 							_firmwareVersion=_storedInputData[1] + _storedInputData[0] / 10;
173 							self.dispatchEvent(new IOBoardEvent(IOBoardEvent.FIRMWARE_VERSION), {version: _firmwareVersion});
174 							break;
175 						case ANALOG_MESSAGE:
176 							var analogPin = self.getAnalogPin(_multiByteChannel);
177 							// NOTE: is there a better way to handle this? This issue is on browser refresh
178 							// the IOBoard board is still sending analog data if analog reporting was set
179 							// before the refresh. Analog reporting won't be disabled by systemReset systemReset()
180 							// is called. There is not way to call that method fast enough so the following code is
181 							// needed. An alternative would be to set a flag that prevents critical operations
182 							// before systemReset has completed
183 							if (analogPin == undefined) {
184 								//console.log("analog pin undefined");
185 								break;
186 							}
187 							// map analog input values from 0-1023 to 0.0 to 1.0
188 							analogPin.value = self.getValueFromTwo7bitBytes(_storedInputData[1], _storedInputData[0])/1023;
189 							if (analogPin.value != analogPin.lastValue) {
190 								// use analog pin number rather than actual pin number
191 								self.dispatchEvent(new IOBoardEvent(IOBoardEvent.ANALOG_DATA), {pin: _multiByteChannel, value: analogPin.value});
192 							}
193 							break;
194 					}
195 				}
196 			}
197 			// we have SysEx command data
198 			else if (_waitForData < 0) {
199 				// we have all sysex data
200 				if (inputData == END_SYSEX) {
201 					_waitForData = 0;
202 					switch (_sysExData[0]) {
203 						case REPORT_FIRMWARE:
204 							processQueryFirmwareResult(_sysExData);
205 							break;
206 						case STRING_DATA:
207 							processSysExString(_sysExData);
208 							break;
209 						case CAPABILITY_RESPONSE:
210 							processCapabilitiesResponse(_sysExData);
211 							break;
212 						case PIN_STATE_RESPONSE:
213 							processPinStateResponse(_sysExData);
214 							break;
215 						case ANALOG_MAPPING_RESPONSE:
216 							processAnalogMappingResponse(_sysExData);
217 							break;
218 						default:
219 							// custom sysEx message
220 							self.dispatchEvent(new IOBoardEvent(IOBoardEvent.SYSEX_MESSAGE), {message: _sysExData});
221 							break;
222 					}
223 					_sysExData = [];
224 				}
225 				else {
226 					_sysExData.push(inputData);
227 				}
228 			}
229 			// we have a command
230 			else {
231 				// extract the command and channel info from a byte if it is less than 0xF0
232 				if (inputData < 240) {
233 					command = inputData & 240;
234 					_multiByteChannel = inputData & 15;
235 				}
236 				else {
237 					// commands in the 0xF* range don't use channel data
238 					command = inputData;
239 				}
240 				switch (command) {
241 					case REPORT_VERSION:
242 					case DIGITAL_MESSAGE:
243 					case ANALOG_MESSAGE:
244 						_waitForData = 2;	// 2 bytes needed
245 						_executeMultiByteCommand = command;
246 						break;
247 					case START_SYSEX:
248 						_waitForData = -1; // n bytes needed
249 						_executeMultiByteCommand = command;
250 						break;
251 					default:
252 						break;
253 				}
254 			}
255 		}

257 		/**
258 		 * @private
259 		 */
260 		function processDigitalPortBytes(port, bits0_6, bits7_13) {
261 			var offset = port * 8;
262 			var lastPin = offset + 8;
263 			var portVal = bits0_6 | (bits7_13 << 7);
264 			var pinVal;
265 			var pin = {};
266 			
267 			if (lastPin >= _totalPins) lastPin = _totalPins;
268 			
269 			var j=0;
270 			for (var i=offset; i<lastPin; i++) {
271 				pin = self.getDigitalPin(i);
272 				// ignore data send on Firmata startup
273 				// same for analog?
274 				if (pin == undefined) return;
275 				
276 				if (pin.type == Pin.DIN) {
277 					pinVal = (portVal >> j) & 0x01;	// test this
278 		    		if (pinVal != pin.value) {
279 		    			pin.value = pinVal;
280 		    			self.dispatchEvent(new IOBoardEvent(IOBoardEvent.DIGITAL_DATA), {pin: pin.number, value: pin.value});
281 		    		}
282 		    	}
283 		    	j++;
284 		    }

318 		/** 
319 		 * Auto configure using capabilities response
320 		 * This should create a configuration for any board in the Firmata boards.h file
321 		 *
322 		 * @private
323 		 */
324 		function processCapabilitiesResponse(msg) {
325 			var pinCapabilities = {};
326 			var byteCounter = 1; // skip 1st byte because it's the command
327 			var pinCounter = 0;
328 			var analogPinCounter = 0;
329 			var len = msg.length;
330 			var type;
331 					
332 			// create default configuration
333 			while (byteCounter <= len) {
334 				// 127 denotes end of pin's modes
335 				if (msg[byteCounter] == 127) {
336 					
337 					// is digital pin mapping even necessary anymore?
338 					_digitalPinMapping[pinCounter] = pinCounter;
339 					type = undefined;
340 					
341 					// assign default types
342 					if (pinCapabilities[Pin.DOUT]) {
343 						// map digital pins
344 						type = Pin.DOUT;
345 					}
346 					
347 					if (pinCapabilities[Pin.AIN]) {
348 						type = Pin.AIN;
349 						// map analog input pins
350 						_analogPinMapping[analogPinCounter++] = pinCounter;
351 					} 
352 					
353 					var pin = new Pin(pinCounter, type);
354 					pin.capabilities = pinCapabilities;
355 					managePinListener(pin);
356 					_ioPins[pinCounter] = pin;
357 					
358 					// store the 2 i2c pin numbers if they exist
359 					if (pin.capabilities[Pin.I2C]) {
360 						_i2cPins.push(pin.number);
361 					}
362 					
363 					pinCapabilities = {};
364 					pinCounter++;
365 					byteCounter++;
366 				} else {
367 					// create capabilities object (mode: resolution) for each  mode
368 					// supported by each pin
369 					pinCapabilities[msg[byteCounter]] = msg[byteCounter + 1];
370 					byteCounter += 2;
371 				}
372 			}
373 			
374 			_numPorts = Math.ceil(pinCounter / 8);
375 			console.log("debug: num ports = " + _numPorts);
376 			
377 			// initialize port values
378 			for (var j=0; j<_numPorts; j++) {
379 				_digitalPort[j] = 0;
380 			}
381 			
382 			_totalPins = pinCounter;
383 			console.log("debug: num pins = " + _totalPins);
384 			
385 			//self.reportCapabilities();
386 
387 			console.log("debug: system reset");
388 			systemReset();
389 
390 			// delay to allow systemReset function to execute in StandardFirmata
391 			setTimeout(startup, 500);
392 			console.log("debug: configured");
393 		}

468 		/**
469 		 * Called when ever a pin value is set via pin.setValue(someValue).
470 		 * Sends digital or analog output pin and output values to the IOBoard.
471 		 *
472 		 * @private
473 		 * @param {Event} pin A reference to the event object (Pin in this case).
474 		 */
475 		 function sendOut(event) {
476 		 	var type = event.target.type;
477 		 	var pinNum = event.target.number;
478 		 	var value = event.target.value;
479 		 	
480 		 	switch(type) {
481 		 		case Pin.DOUT:
482 		 			sendDigitalData(pinNum, value);
483 		 		break;
484 		 		case Pin.AOUT:
485 		 			sendAnalogData(pinNum, value);
486 		 		break;
487 		 		case Pin.SERVO:
488 		 			sendServoData(pinNum, value);
489 		 		break;
490 		 	}
491 
492 		 }

494 		/**
495 		 * Ensure that event listeners are properly managed for pin objects as the pin
496 		 * type is changed during the execution of the program.
497 		 *
498 		 * @private
499 		 */	 
500 		function managePinListener(pin) {
501 			if (pin.type == Pin.DOUT || pin.type == Pin.AOUT || pin.type == Pin.SERVO) {
502 				if (!pin.hasEventListener(Event.CHANGE)) {
503 					pin.addEventListener(Event.CHANGE, sendOut);
504 				}
505 			} else {
506 				if (pin.hasEventListener(Event.CHANGE)) {
507 					try {
508 						pin.removeEventListener(Event.CHANGE, sendOut);
509 					} catch (e) {
510 						// pin had reference to other handler, ignore
511 						console.log("debug: caught pin removeEventListener exception");
512 					}
513 				}
514 			}
515 		}

517 		/**
518 		 * @private
519 		 */
520 		function sendAnalogData(pin, value) {
521 			if (pin > 15 || value > Math.pow(2, 14)) {
522 				sendExtendedAnalogData(pin, value);
523 			} else {
524 				self.send([ANALOG_MESSAGE | (pin & 0x0F), value & 0x007F,(value >> 7) & 0x007F]);
525 			}
526 		}

528 		/**
529 		 * @private
530 		 */	
531 		function sendExtendedAnalogData(pin, value) {
532 			var tempArray = [];
533 			
534 			// if > 16 bits
535 			if (value > Math.pow(2, 16)) {
536 				var err = "Extended Analog values > 16 bits are not currently supported by StandardFirmata";
537 				console.log(err);
538 				throw err;
539 				return;
540 			}
541 			
542 			tempArray[0] = START_SYSEX;
543 			tempArray[1] = EXTENDED_ANALOG;
544 			tempArray[2] = pin;
545 			tempArray[3] = value & 0x007F;
546 		 	tempArray[4] = (value >> 7) & 0x007F;	// up to 14 bits
547 					
548 		 	// if > 14 bits
549 		 	if (value >= Math.pow(2, 14)) {
550 		 		tempArray[5] = (value >> 14) & 0x007F;
551 		 	}
552 		 	
553 			tempArray.push(END_SYSEX);
554 			self.send(tempArray);
555 		}

557 		/**
558 		 * @private
559 		 */
560 		function sendDigitalData(pin, value) {
561 			var portNum = Math.floor(pin / 8);
562 
563 			if (value == Pin.HIGH) {
564 				// set the bit
565 				_digitalPort[portNum] |= (value << (pin % 8));
566 			}
567 			else if (value == Pin.LOW) {
568 				// clear the bit
569 				_digitalPort[portNum] &= ~(1 << (pin % 8));
570 			}
571 			else {
572 				console.log("invalid value passed to sendDigital, value must be 0 or 1");
573 				return; // invalid value
574 			}
575 			
576 			self.sendDigitalPort(portNum, _digitalPort[portNum]);	
577 		}

600 		/**
601 		 * A utility class to assemble a single value from the 2 bytes returned from the
602 		 * IOBoard (since data is passed in 7 bit Bytes rather than 8 bit it must be 
603 		 * reassembled. This is to be used as a protected method and should not be needed
604 		 * in any application level code.
605 		 *
606 		 * @private
607 		 * @param {Number} lsb The least-significant byte of the 2 values to be concatentated
608 		 * @param {Number} msb The most-significant byte of the 2 values to be concatenated
609 		 * @return {Number} The result of merging the 2 bytes
610 		 */
611 		this.getValueFromTwo7bitBytes = function(lsb, msb) {
612 			return (msb <<7) | lsb;
613 		}

799 		 * your javascript message via string messages that you then parse in javascript.
800 		 * You can receive string messages as well.
801 		 *
802 		 * <p>To test, load the EchoString.pde example from Firmata->Examples menu in the
803 		 * IOBoard Application, then use sendString("your string message") to have it
804 		 * echoed back to your javascript application.</p>
805 		 * 
806 		 * @param {String} str The string message to send to the IOBoard
807 		 */
808 		this.sendString = function(str) {

810 			var decValues = [];
811 			for (var i=0, len=str.length; i<len; i++) {

813 				decValues.push((toDec(str[i]) >> 7) & 0x007F);
814 			}
815 			// data > 7 bits in length must be split into 2 bytes and packed into an
816 			// array before passing to the sendSysex method
817 			this.sendSysex(STRING_DATA, decValues);

862 		 * Call to associate a pin with a connected servo motor. See the documentation for your
863 		 * servo motor for the minimum and maximum pulse width. If you can't find it, then the
864 		 * default values should be close enough so call sendServoAttach(pin) omitting the
865 		 * min and max values.
866 		 *
867 		 * @param {Number} pin The pin the server is connected to.
868 		 * @param {Number} minPulse [optional] The minimum pulse width for the servo. Default = 544.
869 		 * @param {Number} maxPulse [optional] The maximum pulse width for the servo. Default = 2400.
870 		 */
871 		this.sendServoAttach = function(pin, minPulse, maxPulse) {
872 			var servoPin;
873 
874 			minPulse = minPulse || 544; 	// default value = 544
875 			maxPulse = maxPulse || 2400;	// default value = 2400
876 		
877 			var tempArray = [];
878 			tempArray[0] = START_SYSEX;
879 			tempArray[1] = SERVO_CONFIG;
880 			tempArray[2] = pin;
881 			tempArray[3] = minPulse % 128;
882 			tempArray[4] = minPulse >> 7;
883 			tempArray[5] = maxPulse % 128;
884 			tempArray[6] = maxPulse >> 7;	
885 			tempArray[7] = END_SYSEX;
886 			
887 			self.send(tempArray);
888
