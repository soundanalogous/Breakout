  6 JSUTILS.namespace('JSUTILS.EventDispatcher');
  7 
  8 JSUTILS.EventDispatcher = (function () {
  9 
 10 	var EventDispatcher;
 11 
 12 	/**
 13 	 * An DOM-like or as3-like EventDispatcher class.
 14 	 *
 15 	 * @class The EventDispatcher class mimics the DOM event dispatcher model so the 
 16 	 * user can add and remove event listeners in a familiar way. Event bubbling is
 17 	 * not available because events are dispatched in relation to state changes of
 18 	 * physical components instead of layered graphics so there is nothing to bubble up.
 19 	 * @exports EventDispatcher as JSUTILS.EventDispatcher
 20 	 * @param {Class} target The instance of the class that implements EventDispatcher
 21 	 */
 22 	EventDispatcher = function(target) {
 23 		"use strict";
 24 		
 25 		this._target = target || null;
 26 		this._eventListeners = {};
 27 		
 28 		this.name = "EventDispatcher";
 29 	};
 30 
 31 	EventDispatcher.prototype = {
 32 
 33 		/**
 34 		 * Description
 35 		 * @param {String} type The event type
 36 		 * @param {Function} listener The function to be called when the event is fired
 37 		 */
 38 		addEventListener: function (type, listener) {
 39 			if(!this._eventListeners[type]) {
 40 				this._eventListeners[type] = [];
 41 			}
 42 			this._eventListeners[type].push(listener);
 43 		},
 44 		
 45 		/**
 46 		 * Description
 47 		 * @param {String} type The event type
 48 		 * @param {Function} listener The function to be called when the event is fired
 49 		 */
 50 		removeEventListener: function(type, listener) {
 51 			for (var i=0, len = this._eventListeners[type].length; i<len; i++) {
 52 				if (this._eventListeners[type][i] === listener) {
 53 					this._eventListeners[type].splice(i, 1);
 54 				}
 55 			}
 56 			// To Do: If no more listeners for a type, delete key?
 57 		},
 58 		
 59 		/**
 60 		 * Description
 61 		 * @param {String} type The event type
 62 		 * return {boolean} True is listener exists for this type, false if not.
 63 		 */
 64 		hasEventListener: function(type) {
 65 			if (this._eventListeners[type] && this._eventListeners[type].length > 0) {
 66 				return true;
 67 			} else {
 68 				return false;
 69 			}	
 70 		},
 71 		
 72 		/**
 73 		 * Description
 74 		 * @param {Event} type The Event object.
 75 		 * @param {Object} optionalParams Optional parameters passed as an object.
 76 		 * return {boolean} True if dispatch is successful, false if not.
 77 		 */	
 78 		dispatchEvent: function(event, optionalParams) {
 79 			
 80 			event.target = this._target;
 81 			var isSuccess = false;		
 82 
 83 			// Add any optional params to the Event object
 84 			for (var obj in optionalParams) {
 85 				event[obj.toString()] = optionalParams[obj];
 86 			}
 87 						
 88 			if (this.hasEventListener(event.type)) {
 89 				for (var j=0, len=this._eventListeners[event.type].length; j<len; j++) {
 90 					try {
 91 						this._eventListeners[event.type][j].call(this, event);
 92 						isSuccess = true;
 93 					} catch(e) {
 94 						// To Do: Handle error
 95 						console.log("error: Error calling event handler. " + e);
 96 					}
 97 				}
 98 			}
 99 			return isSuccess;	
100 		}
101 	};
102 
103 	return EventDispatcher;
104 
105 }());
