  5 BREAKOUT.namespace('BREAKOUT.io.Servo');
  6 
  7 BREAKOUT.io.Servo = (function() {
  8 
  9 	var Servo;
 10 
 11 	// dependencies
 12 	var Pin = BREAKOUT.Pin;
 13 
 14 	/**
 15 	 * Creates a new Servo
 16 	 *
 17 	 * @exports Servo as BREAKOUT.io.Servo
 18 	 * @constructor
 19 	 * @param {IOBoard} board A reference to the IOBoard instance that the servo is attached to.
 20 	 * @param {Pin} servoPin A reference to the Pin the servo is connected to.
 21 	 */
 22 	Servo = function(board, servoPin) {
 23 		"use strict";
 24 		
 25 		this.name = "Servo"; // for testing
 26 
 27 		this._pin = servoPin;
 28 		this._angle;
 29 
 30 		var pinNumber = servoPin.number;
 31 		
 32 		board.sendServoAttach(pinNumber);
 33 	}
 34 
 35 	Servo.prototype = {
 36 
 37 		/**
 38 		 * Set the angle (in degrees) to rotate the server head to.
 39 		 * @name Servo#angle
 40 		 * @property
 41 		 * @type Number
 42 		 */ 
 43 		set angle(value) {
 44 			if (this._pin.getType() == Pin.SERVO) {
 45 				this._angle = value;
 46 				this._pin.value = this._angle;
 47 			}
 48 		},
 49 		get angle() {
 50 			if (this._pin.getType() == Pin.SERVO) {
 51 				return this._angle;
 52 			}
 53 		}		
 54 	};
 55 
 56 	return Servo;
 57 
 58 }());
