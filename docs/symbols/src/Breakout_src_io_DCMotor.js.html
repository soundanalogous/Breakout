 12 JSUTILS.namespace('BO.io.DCMotor');
 13 
 14 BO.io.DCMotor = (function() {
 15 
 16 	var DCMotor;
 17 
 18 	// Dependencies
 19 	var Pin = BO.Pin;
 20 
 21 	/**
 22 	 * H-bridge motor control.
 23 	 * <p>Tested successfully with the following H-bridge: SN754410<br>
 24 	 * Should also be compatible with the following:<br>
 25 	 * SN754410<br>
 26 	 * L293NE<br>
 27 	 * TA7291P<br>
 28 	 * TB6612FNG<br>
 29 	 * BD621F</p>
 30 	 *
 31 	 * @exports DCMotor as BO.io.DCMotor
 32 	 * @class Creates an interface to an H-bridge to control the
 33 	 * direction of rotation of a motor shaft. You can rotate forward
 34 	 * (clockwise), reverse or apply a brake. See 
 35 	 * Breakout/examples/actuators/dcmotor.html for an example application.
 36 	 * @constructor
 37 	 * @param {IOBoard} board A reference to the IOBoard instance that the
 38 	 * servo is attached to.
 39 	 * @param {Pin} forwardPin A reference to the Pin connected to the forward
 40 	 * control of the H-bridge.
 41 	 * @param {Pin} reversePin A reference to the Pin connected to the reverse
 42 	 * control of the H-bridge.
 43 	 * @param {Pin} pwmPin A reference to the Pin connected to the pwm control
 44 	 * of the H-bridge.
 45 	 * @param {Number} minVoltage The minimum voltage (default = 1).
 46 	 * @param {Number} maxVoltage The maximum voltage (default = 9).
 47 	 * @param {Number} supplyVoltage The supply voltage (default = 9).
 48 	 */
 49 	DCMotor = function(board, forwardPin, reversePin, pwmPin, minVoltage, maxVoltage, supplyVoltage) {
 50 		"use strict";
 51 		
 52 		this.name = "DCMotor";
 53 
 54 		minVoltage = minVoltage || 1;
 55 		maxVoltage = maxVoltage || 9;
 56 		supplyVoltage = supplyVoltage || 9;
 57 		if (pwmPin === undefined) pwmPin = null;		
 58 
 59 		this._value = 0;
 60 		this._offset = 0;
 61 		this._range = 0;
 62 
 63 		this._forwardPin = forwardPin;
 64 		this._reversePin = reversePin;
 65 		this._pwmPin = pwmPin;
 66 
 67 		if (this._pwmPin !== null) {
 68 			if (this._pwmPin.getCapabilities()[Pin.PWM]) {
 69 				board.setDigitalPinMode(this._pwmPin.number, Pin.PWM);
 70 			} else {
 71 				console.log("warning: PWM is not available for the PWM pin");
 72 				board.setDigitalPinMode(this._pwmPin.number, Pin.DOUT);
 73 			}			
 74 		}
 75 
 76 		if (this._forwardPin.getCapabilities()[Pin.PWM]) {
 77 			board.setDigitalPinMode(this._forwardPin.number, Pin.PWM);
 78 		} else {
 79 			console.log("warning: PWM is not available for the forward pin");
 80 			board.setDigitalPinMode(this._forwardPin.number, Pin.DOUT);			
 81 		}
 82 
 83 		if (this._reversePin.getCapabilities()[Pin.PWM]) {
 84 			board.setDigitalPinMode(this._reversePin.number, Pin.PWM);			
 85 		} else {
 86 			console.log("warning: PWM is not available for the reverse pin");
 87 			board.setDigitalPinMode(this._reversePin.number, Pin.DOUT);
 88 		}
 89 
 90 		this._offset = (minVoltage / supplyVoltage);
 91 		this._range = (maxVoltage - minVoltage) / supplyVoltage;
 92 
 93 		this.despin(false);
 94 
 95 	};
 96 
 97 	DCMotor.prototype = {
 98 
 99 		/**
100 		 * The value of the motor speed (-1.0 to 1.0). A speed of zero stops
101 		 * the motor.
102 		 * 
103 		 * @name DCMotor#value
104 		 * @property
105 		 * @type Number
106 		 */ 
107 		set value(val) {
108 			this._value = Math.max(-1, Math.min(1, val));
109 
110 			if (val > 0) {
111 				this.forward(this._value);
112 			} else if (val < 0) {
113 				this.reverse(-this._value);
114 			} else {
115 				this.despin();
116 			}
117 		},
118 		get value() {
119 			return this._value;
120 		},
121 		
122 		/**
123 		 * @param {Boolean} useBrake Default = true
124 		 */
125 		despin: function(useBrake) {
126 			if (useBrake === undefined) useBrake = true;
127 
128 			if (useBrake) {
129 				if (this._pwmPin === null) {
130 					this._forwardPin.value = 1;
131 					this._reversePin.value = 1;
132 				} else {
133 					this._forwardPin.value = 1;
134 					this._reversePin.value = 1;
135 					this._pwmPin.value = 1;
136 				}
137 			} else {
138 				if (this._pwmPin ===null) {
139 					this._forwardPin.value = 0;
140 					this._reversePin.value = 0;
141 				} else {
142 					this._forwardPin.value = 0;
143 					this._reversePin.value = 0;
144 					this._pwmPin.value = 0;
145 				}
146 			}
147 			this._value = 0;
148 		},
149 		
150 		/**
151 		 * @param {Number} val The new voltage to set (0.0 to 1.0)
152 		 */		
153 		forward: function(val) {
154 			val = val || 1;
155 			this._value = Math.max(0, Math.min(1, val));
156 
157 			if (this._pwmPin === null) {
158 				this._forwardPin.value = Math.max(0, Math.min(1, this._value * this._range + this._offset));
159 				this._reversePin.value = 0;
160 			} else {
161 				this._forwardPin.value = 1;
162 				this._reversePin.value = 0;
163 				this._pwmPin.value = Math.max(0, Math.min(1, this._value * this._range + this._offset));
164 			}
165 		},
166 
167 		/**
168 		 * @param {Number} val The new voltage to set (-1.0 to 0.0)
169 		 */
170 		reverse: function(val) {
171 			val = val || 1;
172 			this._value = Math.max(0, Math.min(1, val)) * -1;
173 
174 			if (this._pwmPin === null) {
175 				this._forwardPin.value = 0;
176 				this._reversePin.value =  Math.max(0, Math.min(1, (this._value * this._range) * -1 + this._offset));
177 			} else {
178 				this._forwardPin.value = 0;
179 				this._reversePin.value = 1;
180 				this._pwmPin.value = Math.max(0, Math.min(1, (this._value * this._range) * -1 + this._offset));
181 			}
182 		}		
183 			
184 	};
185 
186 	return DCMotor;
187 
188 }());
