 12 BREAKOUT.namespace('BREAKOUT.Pin');
 13 
 14 BREAKOUT.Pin = (function() {
 15 	//"use strict";	// closure compiler doesn't like this... should be fixed soon though
 16 
 17 	var Pin;
 18 
 19 	// dependencies
 20 	var EventDispatcher = BREAKOUT.EventDispatcher,
 21 		Event = BREAKOUT.Event,
 22 		GeneratorEvent = BREAKOUT.generators.GeneratorEvent;
 23 
 24 	/**
 25 	 * An object to represent an IOBoard pin
 26 	 *
 27 	 * @exports Pin as BREAKOUT.Pin
 28 	 * @constructor
 29 	 * @param {Number} number The pin number
 30 	 * @param {Number} type The type of pin
 31 	 */
 32 	Pin = function(number, type) {
 33 
 34 		this.name = "Pin";
 35 
 36 		this._type = type;
 37 		this._capabilities;
 38 		this._number = number;
 39 		this._analogNumber = undefined;
 40 		this._maxPWMValue = 255;
 41 		this._value = -1;
 42 		this._lastValue = -1;
 43 		this._preFilterValue;
 44 		this._average = 0;
 45 		this._minimum = Math.pow(2, 16);
 46 		this._maximum = 0;
 47 		//this._avg = 0;
 48 		this._sum = 0;
 49 		this._numSamples = 0;
 50 		this._filters = null;
 51 		this._generator = null;
 52 		
 53 		this._evtDispatcher = new EventDispatcher(this);	
 54 
 55 	};
 56 
 57 	Pin.prototype = {
 58 		
 59 		/**
 60 		 * The analogNumber sould only be set internally.
 61 		 * @private
 62 		 */		
 63 		setAnalogNumber: function(num) {
 64 			this._analogNumber = num;
 65 		},
 66 
 67 		/**
 68 		 * [read-only] The analog pin number used by the IOBoard (printed on board or datasheet).
 69 		 * @name Pin#analogNumber
 70 		 * @property
 71 		 * @type Number
 72 		 */	
 73 		get analogNumber() {
 74 			return this._analogNumber;
 75 		},		
 76 
 77 		/**
 78 		 * [read-only] The pin number corresponding to the Arduino documentation for the type of board.
 79 		 * @name Pin#number
 80 		 * @property
 81 		 * @type Number
 82 		 */			 
 83 		get number() {
 84 			return this._number;
 85 		},
 86 
 87 		/**
 88 		 * The maximum PWM value supported for this pin. This value should normally
 89 		 * be set internally.
 90 		 * @private
 91 		 */
 92 		setMaxPWMValue: function(val) {
 93 			this._maxPWMValue = value;
 94 		},
 95 
 96 		/**
 97 		 * [read-only] The maximum PWM value supported for this pin.
 98 		 * <p> This is the max PWM value supported by Arduino (currently 255) rather 
 99 		 * than the max PWM value specified by the microcontroller datasheet.</p>
100 		 *
101 		 * @name Pin#maxPWMValue
102 		 * @property
103 		 * @type Number
104 		 */			 
105 		get maxPWMValue() {
106 			return this._maxPWMValue;
107 		},		
108 		
109 		/**
110 		 * [read-only] The average value of the pin over time. Call clear() to reset.
111 		 * @name Pin#average
112 		 * @property
113 		 * @type Number
114 		 */			 
115 		get average() {
116 			return this._average;
117 		},
118 
119 		/**
120 		 * [read-only] The minimum value of the pin over time. Call clear() to reset.
121 		 * @name Pin#minimum
122 		 * @property
123 		 * @type Number
124 		 */			 	
125 		get minimum() {
126 			return this._minimum;
127 		},
128 		
129 		/**
130 		 * [read-only] The maximum value of the pin over time. Call clear() to reset.
131 		 * @name Pin#maximum
132 		 * @property
133 		 * @type Number
134 		 */			 
135 		get maximum() {
136 			return this._maximum;
137 		},
138 		
139 		/**
140 		 * The current digital or analog value of the pin.
141 		 * @name Pin#value
142 		 * @property
143 		 * @type Number
144 		 */		 
145 		get value() {
146 			return this._value;
147 		},
148 		set value(val) {
149 			this.calculateMinMaxAndMean(val);
150 			this._lastValue = this._value;
151 			this._preFilterValue = val;
152 			this._value = this.applyFilters(val);
153 			this.detectChange(this._lastValue, this._value);
154 		},
155 		
156 		/**
157 		 * [read-only] The last pin value.
158 		 * @name Pin#lastValue
159 		 * @property
160 		 * @type Number
161 		 */			 
162 		get lastValue() {
163 			return this._lastValue;
164 		},
165 		
166 		/**
167 		 * [read-only] The value before any filters were applied.
168 		 * @name Pin#preFilterValue
169 		 * @property
170 		 * @type Number
171 		 */			 
172 		get preFilterValue() {
173 			return this._preFilterValue;
174 		},
175 
176 		/**
177 		 * Get and set filters for the Pin.
178 		 * @name Pin#filters
179 		 * @property
180 		 * @type FilterBase
181 		 */	
182 		get filters() {
183 			return this._filters;
184 		},
185 		set filters(filterArray) {
186 			if (filterArray === null || filterArray.length ===0) {
187 				this._filters = filterArray;
188 				return;
189 			}
190 			this._filters = filterArray;
191 		},
192 
193 		/**
194 		 * Get a reference to the current generator.
195 		 * @name Pin#generator
196 		 * @property
197 		 * @type GeneratorBase
198 		 */	
199 		get generator() {
200 			return this._generator;
201 		},
202 
203 		/**
204 		 * The type/mode of the pin (eg. Pin.DIN, Pin.DOUT, etc). Use IOBoard.setDigitalPinMode(pinNumber) 
205 		 * to set the pin type.
206 		 * @return {Number} The pin type/mode
207 		 */	
208 		getType: function() {
209 			return this._type;
210 		},
211 
212 		/**
213 		 * Set the pin type. This method should only be used internaly.
214 		 * @private
215 		 */
216 		setType: function(pinType) {
217 			// ensure pin type is valid
218 			if (pinType >= 0 && pinType < Pin.TOTAL_PIN_MODES) {
219 				this._type = pinType;
220 			}
221 		},			
222 
223 		/**
224 		 * An object storing the capabilities of the pin.
225 		 * @return {Object} An object describing the capabilities of this Pin.
226 		 */	
227 		getCapabilities: function() {
228 			return this._capabilities;
229 		},
230 
231 		/**
232 		 * This method should only be used internally.
233 		 * @private
234 		 */
235 		setCapabilities: function(pinCapabilities) {
236 			this._capabilities = pinCapabilities;
237 		},		
238 
239 		/**
240 		 * Dispatch a Change event whenever a pin value changes
241 		 * @private
242 		 */
243 		detectChange: function(oldValue, newValue) {
244 			if (oldValue === newValue) return;
245 			this.dispatchEvent(new Event(Event.CHANGE));
246 		},
247 		
248 		/**
249 		 * From funnel Pin.as
250 		 * @private
251 		 */
252 		clearWeight: function() {
253 			this._sum = this._average;
254 			this._numSamples = 1;
255 		},
256 		
257 		/**
258 		 * From funnel Pin.as
259 		 * @private
260 		 */
261 		calculateMinMaxAndMean: function(val) {
262 			var MAX_SAMPLES = Number.MAX_VALUE;
263 
264 			this._minimum = Math.min(val, this._minimum);
265 			this._maximum = Math.max(val, this._maximum);
266 			
267 			this._sum += val;
268 			this._average = this._sum / (++this._numSamples);
269 			if (this._numSamples >= MAX_SAMPLES) {
270 				this.clearWeight();
271 			}
272 		},
273 			
274 		/**
275 		 * Resets the minimum, maximum, average and lastValue of the pin.
276 		 */
277 		clear: function() {
278 			this._minimum = this._maximum = this._average = this._lastValue = this._preFilterValue;
279 			this.clearWeight();
280 		},
281 		
282 		/**
283 		 * Add a new filter to the Pin.
284 		 * @param {FilterBase} newFilter A filter object that extends FilterBase.
285 		 */
286 		addFilter: function(newFilter) {
287 
288 			if (newFilter === null) {
289 				return;
290 			}
291 
292 			if (this._filters === null) {
293 				this._filters = [];
294 			}
295 
296 			this._filters.push(newFilter);
297 		},
298 
299 		/**
300 		 * Add a new generator to the Pin. A pin can only have one generator assigned. 
301 		 * Assigning a new generator will replace the previously assigned generator.
302 		 *
303 		 * @param {GeneratorBase} newGenerator A generator object that extends GeneratorBase.
304 		 */
305 		addGenerator: function(newGenerator) {
306 			this.removeGenerator();
307 			this._generator = newGenerator;
308 			this._generator.addEventListener(GeneratorEvent.UPDATE, this.autoSetValue.bind(this));
309 		},
310 
311 		/**
312 		 * Removes the generator from the pin.
313 		 */
314 		removeGenerator: function() {
315 			if (this._generator !== null) {
316 				this._generator.removeEventListener(GeneratorEvent.UPDATE, this.autoSetValue);
317 			}
318 			this._generator = null;				
319 		},
320 
321 		/**
322 		 * Sets new filters to the pin
323 		 * @param {FilterBase[]} newFilters An array of Filter Objects to be applied to the Pin
324 		 */
325 		setFilters: function(newFilters) {
326 			// use the filters setter
327 			this.filters = newFilters;
328 		},
329 
330 		/**
331 		 * Removes all filters from the pin
332 		 */
333 		removeAllFilters: function() {
334 			this._filters = null;
335 		},
336 
337 		/**
338 		 * @private
339 		 */
340 		autoSetValue: function(event) {
341 			var val = this._generator.value;
342 
343 			// scale PWM values to 0 - 255
344 			// if (this._type === Pin.PWM) {
345 			// 	val *= 255.0;
346 			// 	val = (val < 0) ? 0: val;
347 			// 	val = (val > 255) ? 255 : val;
348 			// }
349 			// use the value setter
350 			this.value = val;
351 		},
352 
353 		/**
354 		 * @private
355 		 */
356 		applyFilters: function(val) {
357 			var result;
358 
359 			if (this._filters === null) return val;
360 			
361 			result = val;
362 			var len = this._filters.length;
363 			for (var i = 0; i < len; i++) {
364 				result = this._filters[i].processSample(result);
365 			}
366 
367 			return result;
368 		},
369 
370 		/* implement EventDispatcher */
371 		
372 		/**
373 		 * @param {String} type The event type
374 		 * @param {Function} listener The function to be called when the event is fired
375 		 */
376 		addEventListener: function(type, listener) {
377 			this._evtDispatcher.addEventListener(type, listener);
378 		},
379 		
380 		/**
381 		 * @param {String} type The event type
382 		 * @param {Function} listener The function to be called when the event is fired
383 		 */
384 		removeEventListener: function(type, listener) {
385 			this._evtDispatcher.removeEventListener(type, listener);
386 		},
387 		
388 		/**
389 		 * @param {String} type The event type
390 		 * return {boolean} True is listener exists for this type, false if not.
391 		 */
392 		hasEventListener: function(type) {
393 			return this._evtDispatcher.hasEventListener(type);
394 		},
395 
396 		/**
397 		 * @param {Event} type The Event object
398 		 * @param {Object} optionalParams Optional parameters to assign to the event object.
399 		 * return {boolean} True if dispatch is successful, false if not.
400 		 */	
401 		dispatchEvent: function(event, optionalParams) {
402 			return this._evtDispatcher.dispatchEvent(event, optionalParams);
403 		}		
404 			
405 	};

408 	Pin.HIGH = 1;

412 	Pin.ON = 1;

414 	Pin.OFF = 0;

418 	Pin.DIN = 0x00;
419 	/** @constant */

421 	/** @constant */
422 	Pin.AIN = 0x02;
423 	/** @constant */
424 	Pin.AOUT = 0x03;
425 	/** @constant */
426 	Pin.PWM = 0x03;
427 	/** @constant */
428 	Pin.SERVO = 0x04;
429 	/** @constant */
430 	Pin.SHIFT = 0x05;
431 	/** @constant */
432 	Pin.I2C = 0x06;
433 	/** @constant */

436 	return Pin;

438 }());
