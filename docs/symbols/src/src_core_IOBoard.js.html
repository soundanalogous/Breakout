183 		/**
184 		 * @private
185 		 */
186 	    function processInput(inputData) {
187 	    	inputData *= 1; // Force inputData to integer (is there a better way to do this?)
188 	    	var len;
189 
190 	    	_inputDataBuffer.push(inputData);
191 	    	len = _inputDataBuffer.length;
192 	    	
193 	    	if (_inputDataBuffer[0] >= 128 && _inputDataBuffer[0] != START_SYSEX) {
194 	    		if (len === 3) {
195 	    			processMultiByteCommand(_inputDataBuffer);
196 	    			// Clear buffer
197 	    			_inputDataBuffer = [];
198 	    		}
199 	    	} else if (_inputDataBuffer[0] === START_SYSEX && _inputDataBuffer[len-1] === END_SYSEX) {
200 				processSysexCommand(_inputDataBuffer);
201 	    		// Clear buffer
202 	    		_inputDataBuffer = [];
203 	    	} else if (inputData >= 128 && _inputDataBuffer[0] < 128) {
204 	    		// If for some reason we got a new command and there is already data
205 	    		// in the buffer, reset the buffer
206 	    		console.log("warning: Malformed input data... resetting buffer");
207 	    		_inputDataBuffer = [];
208 	    		if (inputData !== END_SYSEX) {
209 	    			_inputDataBuffer.push(inputData);
210 	    		}
211 	    	}
212 	    }

214 		/**
215 		 * @private
216 		 */	    
217 	    function processMultiByteCommand(commandData) {
218 	    	var command = commandData[0],
219 	    		channel;
220 
221 	    	if (command < 0xF0) {
222 	    		command = command & 0xF0;
223 	    		channel = commandData[0] & 0x0F;
224 	    	}
225 
226 			switch (command) {
227 				case DIGITAL_MESSAGE:
228 					processDigitalMessage(channel, commandData[1], commandData[2]); //(LSB, MSB)
229 					break;
230 				case REPORT_VERSION:
231 					_firmwareVersion = commandData[1] + commandData[2] / 10;
232 					_self.dispatchEvent(new IOBoardEvent(IOBoardEvent.FIRMWARE_VERSION), {version: _firmwareVersion});
233 					break;
234 				case ANALOG_MESSAGE:
235 					processAnalogMessage(channel, commandData[1], commandData[2]);
236 					break;
237 			}
238 	    }

240 		/**
241 		 * @private
242 		 */
243 		function processDigitalMessage(port, bits0_6, bits7_13) {
244 			var offset = port * 8,
245 				lastPin = offset + 8,
246 				portVal = bits0_6 | (bits7_13 << 7),
247 				pinVal,
248 				pin = {};
249 			
250 			if (lastPin >= _totalPins) lastPin = _totalPins;
251 			
252 			var j=0;
253 			for (var i = offset; i < lastPin; i++) {
254 				pin = _self.getDigitalPin(i);
255 				// Ignore data send on Firmata startup
256 				if (pin == undefined) return;
257 				
258 				if (pin.getType() == Pin.DIN) {
259 					pinVal = (portVal >> j) & 0x01;
260 		    		if (pinVal != pin.value) {
261 		    			pin.value = pinVal;
262 		    			_self.dispatchEvent(new IOBoardEvent(IOBoardEvent.DIGITAL_DATA), {pin: pin});
263 		    		}
264 		    	}
265 		    	j++;
266 		    }
267 		}

269 		/**
270 		 * @private
271 		 */
272 	    function processAnalogMessage(channel, bits0_6, bits7_13) {
273 			var analogPin = _self.getAnalogPin(channel);
274 
275 			// NOTE: Is there a better way to handle this? This issue is on
276 			// browser refresh the IOBoard board is still sending analog data
277 			// if analog reporting was set before the refresh. Analog reporting
278 			// won't be disabled by systemReset systemReset() is called. There
279 			// is not a way to call that method fast enough so the following
280 			// code is needed. An alternative would be to set a flag that
281 			// prevents critical operations before systemReset has completed.
282 			if (analogPin === undefined) {
283 				return;
284 			}
285 			// Map analog input values from 0-1023 to 0.0-1.0
286 			// To Do: Add maxADCValue property to Pin or IOBoard object to
287 			// support ADC values > 1023?
288 			// maxADCValue could be set during configuration routine if it's
289 			// supported by Firmata in the future.
290 			analogPin.value = _self.getValueFromTwo7bitBytes(bits0_6, bits7_13)/1023;
291 			if (analogPin.value != analogPin.lastValue) {
292 				_self.dispatchEvent(new IOBoardEvent(IOBoardEvent.ANALOG_DATA), {pin: analogPin});
293 			}	    	
294 	    }

296 		/**
297 		 * @private
298 		 */
299 	    function processSysexCommand(sysexData) {
300 	    	// Remove the first and last element from the array
301 	    	// since these are the START_SYSEX and END_SYSEX 
302 	    	sysexData.shift();
303 	    	sysexData.pop();
304 
305 	    	var command = sysexData[0];
306 			switch (command) {
307 				case REPORT_FIRMWARE:
308 					processQueryFirmwareResult(sysexData);
309 					break;
310 				case STRING_DATA:
311 					processSysExString(sysexData);
312 					break;
313 				case CAPABILITY_RESPONSE:
314 					processCapabilitiesResponse(sysexData);
315 					break;
316 				case PIN_STATE_RESPONSE:
317 					processPinStateResponse(sysexData);
318 					break;
319 				case ANALOG_MAPPING_RESPONSE:
320 					processAnalogMappingResponse(sysexData);
321 					break;
322 				default:
323 					// Custom sysEx message
324 					_self.dispatchEvent(new IOBoardEvent(IOBoardEvent.SYSEX_MESSAGE), {message: sysexData});
325 					break;
326 			}
327 	    }

330 		 * @private

332 		function processQueryFirmwareResult(msg) {
333 			var	data;
334 			for (var i = 3; i < msg.length; i += 2)
335 			{
336 				data = String.fromCharCode(msg[i]);
337 				data += String.fromCharCode(msg[i+1]);
338 				_firmwareName += data;
339 			}
340 			_firmwareVersion = msg[1] + msg[2] / 10;
341 			_self.dispatchEvent(new IOBoardEvent(IOBoardEvent.FIRMWARE_NAME), {name: _firmwareName, version: _firmwareVersion});
342 		}

347 		function processSysExString(msg) {
348 			var str = "",
349 				data,
350 				len = msg.length;

360 		/** 
361 		 * Auto configure using capabilities response.
362 		 * This creates a configuration for any board in the Firmata boards.h 
363 		 * file.
364 		 *
365 		 * @private
366 		 */
367 		function processCapabilitiesResponse(msg) {
368 			// If running in multi-client mode and this client is already 
369 			// configured, ignore capabilities response
370 			if (_isConfigured) return;
371 
372 			var pinCapabilities = {},
373 				byteCounter = 1, // Skip 1st byte because it's the command
374 				pinCounter = 0,
375 				analogPinCounter = 0,
376 				len = msg.length,
377 				type,
378 				pin;
379 					
380 			// Create default configuration
381 			while (byteCounter <= len) {
382 				// 127 denotes end of pin's modes
383 				if (msg[byteCounter] == 127) {
384 					
385 					// Is digital pin mapping even necessary anymore?
386 					_digitalPinMapping[pinCounter] = pinCounter;
387 					type = undefined;
388 					
389 					// Assign default types
390 					if (pinCapabilities[Pin.DOUT]) {
391 						// Map digital pins
392 						type = Pin.DOUT;
393 					}
394 					
395 					if (pinCapabilities[Pin.AIN]) {
396 						type = Pin.AIN;
397 						// Map analog input pins
398 						_analogPinMapping[analogPinCounter++] = pinCounter;
399 					}
400 					
401 					pin = new Pin(pinCounter, type);
402 					pin.setCapabilities(pinCapabilities);
403 					managePinListener(pin);
404 					_ioPins[pinCounter] = pin;
405 					
406 					// Store the 2 i2c pin numbers if they exist
407 					// To Do: allow for more than 2 i2c pins on a board?
408 					// How to identify SDA-SCL pairs in that case?
409 					if (pin.getCapabilities()[Pin.I2C]) {
410 						_i2cPins.push(pin.number);
411 					}
412 					
413 					pinCapabilities = {};
414 					pinCounter++;
415 					byteCounter++;
416 				} else {
417 					// Create capabilities object (mode: resolution) for each 
418 					// mode supported by each pin
419 					pinCapabilities[msg[byteCounter]] = msg[byteCounter + 1];
420 					byteCounter += 2;
421 				}
422 			}
423 			
424 			_numPorts = Math.ceil(pinCounter / 8);
425 			debug("debug: Num ports = " + _numPorts);
426 			
427 			// Initialize port values
428 			for (var j = 0; j < _numPorts; j++) {
429 				_digitalPort[j] = 0;
430 			}
431 			
432 			_totalPins = pinCounter;
433 			debug("debug: Num pins = " + _totalPins);
434 
435 			// Map the analog pins to the board pins
436 			// This will map the IOBoard analog pin numbers (printed on IOBoard)
437 			// to their digital pin number equivalents
438 			queryAnalogMapping();
439 		}

441 		/**
442 		 * Map map analog pins to board pin numbers. Need to do this because
443 		 * the capability query does not provide the correct order of analog
444 		 * pins.
445 		 *
446 		 * @private
447 		 */
448 		function processAnalogMappingResponse(msg) {
449 			// If running in multi-client mode and this client is 
450 			// already configured ignore analog mapping response
451 			if (_isConfigured) return;
452 
453 			var len = msg.length;
454 			for (var i = 1; i < len; i++) {
455 				if (msg[i] != 127) {
456 					_analogPinMapping[msg[i]] = i - 1;
457 					_self.getPin(i-1).setAnalogNumber(msg[i]);
458 				}
459 			}
460 			
461 			if (!_isMultiClientEnabled) {
462 				startupInSingleClientMode();
463 			} else {
464 				startupInMultiClientMode()
465 			}
466 		}

525 		 * @private

527 		function processPinStateResponse(msg) {
528 			// If running in multi-client mode and this client is 
529 			// already configured ignore pin state response
530 			if (_isConfigured) return;
531 						
532 			var len = msg.length,
533 				pinNumber = msg[1],
534 				pinType = msg[2],
535 				value,
536 				pin = _ioPins[pinNumber];
537 
538 			if (len > 4) {
539 				// Get value
540 				value = _self.getValueFromTwo7bitBytes(msg[3], msg[4]);
541 			} else if (len > 3) {
542 				value = msg[3];
543 			}
544 			
545 			if (pin.getType() != pinType) {
546 				pin.setType(pinType);
547 				managePinListener(pin);
548 			}
549 			if (pin.value != value) {
550 				pin.value = value;
551 			}
552 			
553 			_self.dispatchEvent(new IOBoardEvent(IOBoardEvent.PIN_STATE_RESPONSE), {pin: pin});
554 		}

567 		/**
568 		 * Called when ever a pin value is set via pin.value = someValue.
569 		 * Sends digital or analog output pin and output values to the IOBoard.
570 		 *
571 		 * @private
572 		 * @param {Event} event A reference to the event object (Pin in this
573 		 * case).
574 		 */
575 		 function sendOut(event) {
576 		 	var type = event.target.getType(),
577 		 		pinNum = event.target.number,
578 		 		value = event.target.value;
579 		 	
580 		 	switch(type) {
581 		 		case Pin.DOUT:
582 		 			sendDigitalData(pinNum, value);
583 		 		break;
584 		 		case Pin.AOUT:
585 		 			sendAnalogData(pinNum, value);
586 		 		break;
587 		 		case Pin.SERVO:
588 		 			sendServoData(pinNum, value);
589 		 		break;
590 		 	}
591 		 }

593 		/**
594 		 * Ensure that event listeners are properly managed for pin objects 
595 		 * as the pin type is changed during the execution of the program.
596 		 *
597 		 * @private
598 		 */	 
599 		function managePinListener(pin) {
600 			if (pin.getType() == Pin.DOUT || pin.getType() == Pin.AOUT || pin.getType() == Pin.SERVO) {
601 				if (!pin.hasEventListener(PinEvent.CHANGE)) {
602 					pin.addEventListener(PinEvent.CHANGE, sendOut);
603 				}
604 			} else {
605 				if (pin.hasEventListener(PinEvent.CHANGE)) {
606 					try {
607 						pin.removeEventListener(PinEvent.CHANGE, sendOut);
608 					} catch (e) {
609 						// Pin had reference to other handler, ignore
610 						debug("debug: Caught pin removeEventListener exception");
611 					}
612 				}
613 			}
614 		}

616 		/**
617 		 * @private
618 		 */
619 		function sendAnalogData(pin, value) {
620 
621 			var pwmMax = _self.getDigitalPin(pin).maxPWMValue;
622 			value *= pwmMax;
623 			value = (value < 0) ? 0: value;
624 			value = (value > pwmMax) ? pwmMax : value;
625 
626 			if (pin > 15 || value > Math.pow(2, 14)) {
627 				sendExtendedAnalogData(pin, value);
628 			} else {
629 				_self.send([ANALOG_MESSAGE | (pin & 0x0F), value & 0x007F,(value >> 7) & 0x007F]);
630 			}
631 		}

633 		/**
634 		 * @private
635 		 */	
636 		function sendExtendedAnalogData(pin, value) {
637 			var analogData = [];
638 			
639 			// If > 16 bits
640 			if (value > Math.pow(2, 16)) {
641 				var err = "error: Extended Analog values > 16 bits are not currently supported by StandardFirmata";
642 				console.log(err);
643 				throw err;
644 			}
645 			
646 			analogData[0] = START_SYSEX;
647 			analogData[1] = EXTENDED_ANALOG;
648 			analogData[2] = pin;
649 			analogData[3] = value & 0x007F;
650 		 	analogData[4] = (value >> 7) & 0x007F;	// Up to 14 bits
651 					
652 		 	// If > 14 bits
653 		 	if (value >= Math.pow(2, 14)) {
654 		 		analogData[5] = (value >> 14) & 0x007F;
655 		 	}
656 		 	
657 			analogData.push(END_SYSEX);
658 			_self.send(analogData);
659 		}

661 		/**
662 		 * @private
663 		 */
664 		function sendDigitalData(pin, value) {
665 			var portNum = Math.floor(pin / 8);
666 
667 			if (value == Pin.HIGH) {
668 				// Set the bit
669 				_digitalPort[portNum] |= (value << (pin % 8));
670 			}
671 			else if (value == Pin.LOW) {
672 				// Clear the bit
673 				_digitalPort[portNum] &= ~(1 << (pin % 8));
674 			}
675 			else {
676 				console.log("warning: Invalid value passed to sendDigital, value must be 0 or 1.");
677 				return; // Invalid value
678 			}
679 			
680 			_self.sendDigitalPort(portNum, _digitalPort[portNum]);	
681 		}
