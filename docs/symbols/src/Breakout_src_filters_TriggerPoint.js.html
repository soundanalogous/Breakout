 12 JSUTILS.namespace('BO.filters.TriggerPoint');
 13 
 14 BO.filters.TriggerPoint = (function() {
 15 	"use strict";
 16 
 17 	var TriggerPoint;
 18 
 19 	// dependencies
 20 	var FilterBase = BO.filters.FilterBase;
 21 
 22 	/**
 23 	 * Divides an input to 0 or 1 based on the threshold and hysteresis. You can also
 24 	 * use multiple points by providing a nested array such as [[0.4, 0.1], [0.7, 0.05]].
 25 	 *
 26 	 * @exports TriggerPoint as BO.filters.TriggerPoint
 27 	 * @class Divides an input to 0 or 1 based on the threshold and hysteresis. You can also
 28 	 * use multiple points by providing a nested array such as [[0.4, 0.1], [0.7, 0.05]].
 29 	 * See Breakout/examples/filters/triggerpoint.html for an example application.
 30 	 * @constructor
 31 	 * @augments BO.filters.FilterBase
 32 	 * @param {Number[]} points An array of threshold and hysteresis values
 33 	 * operations for input buffers.
 34 	 */
 35 	TriggerPoint = function(points) {
 36 
 37 		this.name = "TriggerPoint";
 38 
 39 		this._points = {};
 40 		this._range = [];
 41 		this._lastStatus;
 42 
 43 		if (points === undefined) points = [[0.5, 0]];
 44 
 45 		if (points[0] instanceof Array) {
 46 			var len = points.length;
 47 			for (var i=0; i<len; i++) {
 48 				this._points[points[i][0]] = points[i][1];	
 49 			}
 50 		} else if (typeof points[0] === "number") {
 51 			this._points[points[0]] = points[1];
 52 		}
 53 
 54 		this.updateRange();
 55 
 56 		this._lastStatus = 0;
 57 	};
 58 
 59 
 60 	TriggerPoint.prototype = JSUTILS.inherit(FilterBase.prototype);
 61 	TriggerPoint.prototype.constructor = TriggerPoint;
 62 
 63 	/**
 64 	 * Override FilterBase.processSample
 65 	 *
 66 	 * @inheritDoc
 67 	 */
 68 	TriggerPoint.prototype.processSample = function(val) {
 69 		var status = this._lastStatus;
 70 		var len = this._range.length;
 71 		for (var i=0; i<len; i++) {
 72 			var range = this._range[i];
 73 			if (range[0] <= val && val <= range[1]) {
 74 				status = i;
 75 				break;
 76 			}
 77 		}
 78 
 79 		this._lastStatus = status;
 80 		return status;
 81 	};
 82 
 83 	TriggerPoint.prototype.addPoint = function(threshold, hysteresis) {
 84 		this._points[threshold] = hysteresis;
 85 		this.updateRange();
 86 	};
 87 
 88 	TriggerPoint.prototype.removePoint = function(threshold) {
 89 		// to do: verify that this works in javascript
 90 		delete this._points[threshold];
 91 		this.updateRange();
 92 	};
 93 
 94 	TriggerPoint.prototype.removeAllPoints = function() {
 95 		this._points = {};
 96 		this.updateRange();
 97 	};
 98 
 99 	/**
100 	 * @private
101 	 */
102 	TriggerPoint.prototype.updateRange = function() {
103 				
104 		this._range = [];
105 		var keys = this.getKeys(this._points);
106 
107 		var firstKey = keys[0];
108 		this._range.push([Number.NEGATIVE_INFINITY, firstKey - this._points[firstKey]]);
109 
110 		var len = keys.length - 1;
111 		for (var i=0; i<len; i++) {
112 			var t0 = keys[i];
113 			var t1 = keys[i+1];
114 			var p0 = (t0 * 1) + this._points[t0]; // multiply by 1 to force type to number
115 			var p1 = t1 - this._points[t1];
116 			if (p0 >= p1) throw new Error("The specified range overlaps...");
117 			this._range.push([p0, p1]);
118 		}
119 
120 		var lastKey = keys[keys.length - 1];
121 		var positiveThresh = (lastKey * 1) + this._points[lastKey];
122 		this._range.push([positiveThresh, Number.POSITIVE_INFINITY]);
123 
124 	};
125 
126 	/**
127 	 * @private
128 	 */
129 	TriggerPoint.prototype.getKeys = function(obj) {
130 		var keys = [];
131 		for (var key in obj) {
132 			keys.push(key);
133 		}
134 		return keys.sort();
135 	};
136 		
137 	return TriggerPoint;
138 
139 }());
