  6 JSUTILS.namespace('BO.io.Stepper');
  7 
  8 BO.io.Stepper = (function () {
  9     "use strict";
 10 
 11     var Stepper;
 12     var instanceCounter = 0;
 13 
 14     // private static constants
 15     var STEPPER = 0x72,
 16         CONFIG = 0,
 17         STEP = 1,
 18         MAX_STEPS = 2097151, // 21 bits (2^21 - 1)
 19         MAX_SPEED = 16383; // 14 bits (2^14 - 1)
 20 
 21     // dependencies
 22     var Pin = BO.Pin,
 23         EventDispatcher = JSUTILS.EventDispatcher,
 24         Event = JSUTILS.Event,
 25         IOBoardEvent = BO.IOBoardEvent;
 26 
 27     /**
 28      * Creates a new Stepper.
 29      *
 30      * @exports Stepper as BO.io.Stepper
 31      * @class Creates an interface to a Stepper motor. Use this object to set
 32      * the direction and number of steps for the motor to rotate. See
 33      * Breakout/examples/actuators/stepper_2wire.html, stepper_4wire.html, 
 34      * stepper_easydriver.html and stepper_simple.html for example applications.
 35      *
 36      * @constructor
 37      * @param {IOBoard} board A reference to the IOBoard instance that the 
 38      * stepper is attached to.
 39      * @param {Number} driverType. The type of driver (Stepper.DRIVER, 
 40      * Stepper.TWO_WIRE, or Stepper.FOUR_WIRE).
 41      * @param {Number} numStepsPerRev The number of steps to make 1 revolution. 
 42      * @param {Pin} directionPin If dirver interface, the pin used to control 
 43      * the direction.
 44      * If 2-wire or 4-wire interface, the 1st moter pin.
 45      * @param {Pin} stepPin If dirver interface, the pin used to control the 
 46      * steps.
 47      * If 2-wire or 4-wire interface, the 2nd moter pin.
 48      * @param {Pin} motorPin3 [optional] Only required for a 4-wire interface.
 49      * @param {Pin} motorPin4 [optional] Only required for a 4-wire interface.
 50      *
 51      * @example
 52      * // example
 53      * var Stepper = BO.io.Stepper,
 54      *     Event = JSUTILS.Event;
 55      *
 56      * var stepper,
 57      *     stepsPerRev = 200,           // update this for your stepper
 58      *     numSteps = stepsPerRev * 10, // 10 revolutions (+ CW, - CCW)
 59      *     speed = 15.0,                // rad/sec (RPM = speed * 9.55)
 60      *     acceleration = 20.0,         // rad/sec^2
 61      *     deceleration = 20.0;         // rad/sec^2
 62      *
 63      * stepper = new Stepper(arduino,
 64      *              Stepper.TWO_WIRE, // or Stepper.DRIVER or Stepper.FOUR_WIRE
 65      *              stepsPerRev,
 66      *              arduino.getDigitalPin(2),
 67      *              arduino.getDigitalPin(3));
 68      *
 69      * stepper.addEventListener(Event.COMPLETE, onStepperComplete);
 70      *
 71      * // acceleration and deceleration parameters are optional
 72      * stepper.step(numSteps, speed, acceleration, deceleration);
 73      *
 74      * function onStepperComplete(event) {
 75      *     // each stepper is assigned a read-only id value when instantiated
 76      *     console.log("stepper " + event.target.id + " sequence complete");
 77      * }
 78      */
 79     Stepper = function (board, driverType, numStepsPerRev, directionPin, stepPin, motorPin3, motorPin4) {
 80         
 81         // create a new id each time a new instance is created
 82         this._id = instanceCounter++;
 83         if (this._id > 5) {
 84             console.log("Warning: A maximum of 6 Stepper instances can be created");
 85             return;
 86         }
 87 
 88         this.name = "Stepper";
 89         this._board = board;
 90         this._evtDispatcher = new EventDispatcher(this);
 91 
 92         var numStepsPerRevLSB = numStepsPerRev & 0x007F,
 93             numStepsPerRevMSB = (numStepsPerRev >> 7) & 0x007F;
 94 
 95         this._board.setDigitalPinMode(directionPin.number, Pin.DOUT);
 96         this._board.setDigitalPinMode(stepPin.number, Pin.DOUT);
 97 
 98         this._board.addEventListener(IOBoardEvent.SYSEX_MESSAGE, this.onSysExMessage.bind(this));
 99         
100         switch (driverType) {
101         case Stepper.DRIVER:
102         case Stepper.TWO_WIRE:
103             // configure the stepper motor
104             this._board.sendSysex(STEPPER,
105                                     [CONFIG,
106                                     this._id,
107                                     driverType,
108                                     numStepsPerRevLSB, 
109                                     numStepsPerRevMSB,
110                                     directionPin.number,
111                                     stepPin.number]);
112             break;
113         case Stepper.FOUR_WIRE:
114             this._board.setDigitalPinMode(motorPin3.number, Pin.DOUT);
115             this._board.setDigitalPinMode(motorPin4.number, Pin.DOUT);
116 
117             // configure the stepper motor
118             this._board.sendSysex(STEPPER,
119                                     [CONFIG,
120                                     this._id,
121                                     driverType,
122                                     numStepsPerRevLSB, 
123                                     numStepsPerRevMSB,
124                                     directionPin.number,
125                                     stepPin.number,
126                                     motorPin3.number,
127                                     motorPin4.number]);
128             break;
129         }
130 
131     };
132 
133     Stepper.prototype = {
134 
135         constructor: Stepper,
136 
137         /**
138          * Move the stepper a given number of steps at the specified
139          * speed (rad/sec), acceleration (rad/sec^2) and deceleration (rad/sec^2).
140          * The accel and decel parameters are optional but if using, both values
141          * must be passed to the function.
142          *
143          * @param {Number} numSteps The number ofsteps to move the motor (max = +/-2097151 (21 bits)).
144          * Positive value is clockwise, negative value is counter clockwise.
145          * @param {Number} speed Max speed in rad/sec (1 rad/sec = 9.549 RPM)
146          * (max precision of 2 decimal places)
147          * @param {Number} accel [optional] Acceleration in rad/sec^2 (max precision of 2 decimal places)
148          * @param {Number} decel [optional] Deceleration in rad/sec^2 (max precision of 2 decimal places)
149          */      
150         step: function (numSteps, speed, accel, decel) {
151             var steps,
152                 speedLSB,
153                 speedMSB,
154                 accelLSB,
155                 accelMSB,
156                 decelLSB,
157                 decelMSB,
158                 direction = Stepper.CLOCKWISE;
159 
160             if (numSteps > MAX_STEPS) {
161                 numSteps = MAX_STEPS;
162                 console.log("Warning: Maximum number of steps (2097151) exceeded. Setting to step number to 2097151");
163             }
164             if (numSteps < -MAX_STEPS) {
165                 numSteps = -MAX_STEPS;
166                 console.log("Warning: Maximum number of steps (-2097151) exceeded. Setting to step number to -2097151");
167             }
168 
169             if (numSteps > 0) {
170                 direction = Stepper.COUNTER_CLOCKWISE;
171             }
172 
173             steps = [
174                 Math.abs(numSteps) & 0x0000007F,
175                 (Math.abs(numSteps) >> 7) & 0x0000007F,
176                 (Math.abs(numSteps) >> 14) & 0x0000007F
177             ];
178 
179             // the stepper interface expects decimal expressed an an integer
180             speed = Math.floor(speed.toFixed(2) * 100);
181 
182             if (speed > MAX_SPEED) {
183                 speed = MAX_SPEED;
184                 console.log("Warning: Maximum speed (163.83 rad/sec) exceeded. Setting speed to 163.83 rad/sec");
185             }                   
186 
187             speedLSB = speed & 0x007F;
188             speedMSB = (speed >> 7) & 0x007F;
189 
190             // make sure both accel and decel are defined
191             if (accel !== undefined && decel !== undefined) {
192                 // the stepper interface expects decimal expressed an an integer
193                 accel = Math.floor(accel.toFixed(2) * 100);
194                 decel = Math.floor(decel.toFixed(2) * 100);
195 
196                 accelLSB = accel & 0x007F;
197                 accelMSB = (accel >> 7) & 0x007F;
198 
199                 decelLSB = decel & 0x007F;
200                 decelMSB = (decel >> 7) & 0x007F;               
201                             
202                 this._board.sendSysex(STEPPER, 
203                                         [STEP,
204                                         this._id,
205                                         direction,                              
206                                         steps[0],
207                                         steps[1],
208                                         steps[2],
209                                         speedLSB,
210                                         speedMSB,
211                                         accelLSB,
212                                         accelMSB,
213                                         decelLSB,
214                                         decelMSB
215                                         ]);
216             } else {
217                 // don't send accel and decel values
218                 this._board.sendSysex(STEPPER, 
219                                         [STEP,
220                                         this._id,
221                                         direction,                                  
222                                         steps[0],
223                                         steps[1],
224                                         steps[2],
225                                         speedLSB,
226                                         speedMSB
227                                         ]);             
228             }
229         },
230 
231         /**
232          * Listen for stepping complete event
233          *
234          * @private
235          */
236         onSysExMessage: function (event) {
237             var message = event.message;
238 
239             if (message[0] !== STEPPER) {
240                 return;
241             } else if (message[1] !== this._id) {
242                 return;
243             } else {
244                 this.dispatchEvent(new Event(Event.COMPLETE));
245             }
246         },
247 
248         /**
249          * @return {Number} The id of the Stepper object instance
250          */
251         get id() {
252             return this._id;
253         },
254 
255         /* implement EventDispatcher */
256         
257         /**
258          * @param {String} type The event type
259          * @param {Function} listener The function to be called when the event is fired
260          */
261         addEventListener: function (type, listener) {
262             this._evtDispatcher.addEventListener(type, listener);
263         },
264         
265         /**
266          * @param {String} type The event type
267          * @param {Function} listener The function to be called when the event is fired
268          */
269         removeEventListener: function (type, listener) {
270             this._evtDispatcher.removeEventListener(type, listener);
271         },
272         
273         /**
274          * @param {String} type The event type
275          * return {boolean} True is listener exists for this type, false if not.
276          */
277         hasEventListener: function (type) {
278             return this._evtDispatcher.hasEventListener(type);
279         },
280         
281         /**
282          * @param {Event} type The Event object
283          * @param {Object} optionalParams Optional parameters to assign to the event object.
284          * return {boolean} True if dispatch is successful, false if not.
285          */     
286         dispatchEvent: function (event, optionalParams) {
287             return this._evtDispatcher.dispatchEvent(event, optionalParams);
288         }           
289     
290     };
291 
292     /** @constant */
293     Stepper.CLOCKWISE = 0;
294     /** @constant */
295     Stepper.COUNTER_CLOCKWISE = 1;
296     /** @constant */
297     Stepper.DRIVER = 1;
298     /** @constant */
299     Stepper.TWO_WIRE = 2;
300     /** @constant */
301     Stepper.FOUR_WIRE = 4;              
302 
303     return Stepper;
304 
305 })();
