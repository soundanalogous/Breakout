  6 JSUTILS.namespace('BO.io.Stepper');
  7 
  8 BO.io.Stepper = (function() {
  9 	"use strict";
 10 
 11 	var Stepper;
 12 	var instanceCounter = 0;
 13 
 14 	// private static constants
 15 	var CONFIG = 0,
 16 		STEP = 1,
 17 		MAX_STEPS = 2097151, // 21 bits (2^21 - 1)
 18 		MAX_SPEED = 16383;
 19 
 20 	// dependencies
 21 	var Pin = BO.Pin,
 22 		EventDispatcher = JSUTILS.EventDispatcher,
 23 		Event = JSUTILS.Event,
 24 		IOBoardEvent = BO.IOBoardEvent;
 25 
 26 	/**
 27 	 * Creates a new Stepper.
 28 	 *
 29 	 * @exports Stepper as BO.io.Stepper
 30 	 * @class Creates an interface to a Stepper motor. Use this object to set
 31 	 * the direction and number of steps for the motor to rotate. See
 32 	 * Breakout/examples/actuators/stepper.html for an example application.
 33 	 * @constructor
 34 	 * @param {IOBoard} board A reference to the IOBoard instance that the 
 35 	 * servo is attached to.
 36 	 * @param {Number} driverType. The type of driver (Stepper.DRIVER, 
 37 	 * Stepper.TWO_WIRE, or Stepper.FOUR_WIRE).
 38 	 * @param {Number} numStepsPerRev The number of steps to make 1 revolution. 
 39 	 * @param {Pin} directionPin If dirver interface, the pin used to control the direction.
 40 	 * If 2-wire interface, the 1st moter pin.
 41 	 * @param {Pin} stepPin If dirver interface, the pin used to control the steps.
 42 	 * If 2-wire interface, the 2nd moter pin.
 43 	 * @param {Pin} motorPin3 [optional] Only required for a 4-wire interface.
 44 	 * @param {Pin} motorPin4 [optional] Only required for a 4-wire interface.	 	 
 45 	 */
 46 	Stepper = function(board, driverType, numStepsPerRev, directionPin, stepPin, motorPin3, motorPin4) {
 47 		"use strict";
 48 		
 49 		// create a new id each time a new instance is created
 50 		this._id = instanceCounter++;
 51 		if (this._id > 5) {
 52 			console.log("Warning: A maximum of 6 Stepper instances can be created");
 53 			return;
 54 		}
 55 
 56 		this.name = "Stepper";
 57 		this._board = board;
 58 		this._evtDispatcher = new EventDispatcher(this);
 59 
 60 		var numStepsPerRevLSB = numStepsPerRev & 0x007F,
 61 			numStepsPerRevMSB = (numStepsPerRev >> 7) & 0x007F;
 62 
 63 		this._board.setDigitalPinMode(directionPin.number, Pin.DOUT);
 64 		this._board.setDigitalPinMode(stepPin.number, Pin.DOUT);
 65 
 66 		this._board.addEventListener(IOBoardEvent.SYSEX_MESSAGE, this.onSysExMessage.bind(this));
 67 		
 68 		switch (driverType) {
 69 			case Stepper.DRIVER:
 70 			case Stepper.TWO_WIRE:
 71 				// configure the stepper motor
 72 				this._board.sendSysex(Stepper.STEPPER,
 73 												[CONFIG,
 74 												this._id,
 75 												driverType,
 76 												numStepsPerRevLSB, 
 77 												numStepsPerRevMSB,
 78 												directionPin.number,
 79 												stepPin.number]);
 80 				break;
 81 			case Stepper.FOUR_WIRE:
 82 				this._board.setDigitalPinMode(motorPin3.number, Pin.DOUT);
 83 				this._board.setDigitalPinMode(motorPin4.number, Pin.DOUT);
 84 
 85 				// configure the stepper motor
 86 				this._board.sendSysex(Stepper.STEPPER,
 87 												[CONFIG,
 88 												this._id,
 89 												driverType,
 90 												numStepsPerRevLSB, 
 91 												numStepsPerRevMSB,
 92 												directionPin.number,
 93 												stepPin.number,
 94 												motorPin3.number,
 95 												motorPin4.number]);
 96 				break;
 97 		}
 98 
 99 	};
100 
101 	Stepper.prototype = {
102 
103 		/**
104 		 * Move the stepper a given number of steps at the specified
105 		 * speed (rad/sec), acceleration (rad/sec^2) and deceleration (rad/sec^2).
106 		 *
107 		 * @param {Number} numSteps The number ofsteps to move the motor (max = +/-2097151 (21 bits)).
108 		 * Positive value is clockwise, negative value is counter clockwise.
109 		 * @param {Number} speed Max speed in 0.01*rad/sec 
110 		 * @param {Number} accel [optional] Acceleration in 0.01*rad/sec^2
111 		 * @param {Number} decel [optional] Deceleration in 0.01*rad/sec^2
112 		 */		 
113 		step: function(numSteps, speed, accel, decel) {
114 			var steps,
115 				speedLSB,
116 				speedMSB,
117 				accelLSB,
118 				accelMSB,
119 				decelLSB,
120 				decelMSB,
121 				direction = Stepper.CLOCKWISE;
122 
123 			if (numSteps > MAX_STEPS) {
124 				numSteps = MAX_STEPS;
125 				console.log("Warning: Maximum number of steps (2097151) exceeded. Setting to step number to 2097151");
126 			}
127 			if (numSteps < -MAX_STEPS) {
128 				numSteps = -MAX_STEPS;
129 				console.log("Warning: Maximum number of steps (-2097151) exceeded. Setting to step number to -2097151");
130 			}
131 
132 			if (numSteps > 0) {
133 				direction = Stepper.COUNTER_CLOCKWISE;
134 			}
135 
136 			steps = [
137 				Math.abs(numSteps) & 0x0000007F,
138 				(Math.abs(numSteps) >> 7) & 0x0000007F,
139 				(Math.abs(numSteps) >> 14) & 0x0000007F
140 				];
141 
142 			if (speed > MAX_SPEED) {
143 				speed = MAX_SPEED;
144 				// TO DO: determin what the absolute max is when using stepper with Firmata.
145 				// It's likely far less that 16384 rad/sec
146 				console.log("Warning: Maximum speed (16,383) exceeded. Setting speed to 16,383 RPM");
147 			}					
148 
149 			speedLSB = speed & 0x007F;
150 			speedMSB = (speed >> 7) & 0x007F;
151 
152 			// make sure both accel and decel are defined
153 			if (accel !== undefined && decel !== undefined) {
154 				accelLSB = accel & 0x007F;
155 				accelMSB = (accel >> 7) & 0x007F;
156 
157 				decelLSB = decel & 0x007F;
158 				decelMSB = (decel >> 7) & 0x007F;				
159 							
160 				this._board.sendSysex(Stepper.STEPPER, 
161 												[STEP,
162 												this._id,
163 												direction,								
164 												steps[0],
165 												steps[1],
166 												steps[2],
167 												speedLSB,
168 												speedMSB,
169 												accelLSB,
170 												accelMSB,
171 												decelLSB,
172 												decelMSB
173 												]);
174 			} else {
175 				// don't send accel and decel values
176 				this._board.sendSysex(Stepper.STEPPER, 
177 												[STEP,
178 												this._id,
179 												direction,									
180 												steps[0],
181 												steps[1],
182 												steps[2],
183 												speedLSB,
184 												speedMSB
185 												]);				
186 			}
187 		},
188 
189 		/**
190 		 * Listen for stepping complete event
191 		 *
192 		 * @private
193 		 */
194 		onSysExMessage: function(event) {
195 			var message = event.message;
196 
197 			if (message[0] !== Stepper.STEPPER) {
198 				return;
199 			} else if (message[1] !== this._id) {
200 				return;
201 			} else {
202 				this.dispatchEvent(new Event(Event.COMPLETE));
203 			}
204 		},
205 
206 		/* implement EventDispatcher */
207 		
208 		/**
209 		 * @param {String} type The event type
210 		 * @param {Function} listener The function to be called when the event is fired
211 		 */
212 		addEventListener: function(type, listener) {
213 			this._evtDispatcher.addEventListener(type, listener);
214 		},
215 		
216 		/**
217 		 * @param {String} type The event type
218 		 * @param {Function} listener The function to be called when the event is fired
219 		 */
220 		removeEventListener: function(type, listener) {
221 			this._evtDispatcher.removeEventListener(type, listener);
222 		},
223 		
224 		/**
225 		 * @param {String} type The event type
226 		 * return {boolean} True is listener exists for this type, false if not.
227 		 */
228 		hasEventListener: function(type) {
229 			return this._evtDispatcher.hasEventListener(type);
230 		},
231 		
232 		/**
233 		 * @param {Event} type The Event object
234 		 * @param {Object} optionalParams Optional parameters to assign to the event object.
235 		 * return {boolean} True if dispatch is successful, false if not.
236 		 */		
237 		dispatchEvent: function(event, optionalParams) {
238 			return this._evtDispatcher.dispatchEvent(event, optionalParams);
239 		}			
240 	
241 	};
242 
243 	/** @constant */
244 	Stepper.STEPPER	= 0x72;
245 	/** @constant */
246 	Stepper.CLOCKWISE = 0;
247 	/** @constant */
248 	Stepper.COUNTER_CLOCKWISE = 1;
249 	/** @constant */
250 	Stepper.DRIVER = 1;
251 	/** @constant */
252 	Stepper.TWO_WIRE = 2;
253 	/** @constant */
254 	Stepper.FOUR_WIRE = 4;				
255 
256 	return Stepper;
257 
258 })();
