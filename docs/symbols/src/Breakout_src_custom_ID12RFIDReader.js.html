  6 JSUTILS.namespace('BO.custom.RFIDEvent');
  7 
  8 BO.custom.RFIDEvent = (function() {
  9 	"use strict";
 10 
 11 	var RFIDEvent;
 12 
 13 	// dependencies
 14 	var Event = JSUTILS.Event;
 15 
 16 	/**
 17 	 * @exports RFIDEvent as BO.custom.RFIDEvent
 18 	 * @constructor
 19 	 * @augments JSUTILS.Event
 20 	 * @param {String} type The event type
 21 	 * @param {String} tag The RFID tag value (hexadecimal)
 22 	 */
 23 	RFIDEvent = function(type, tag) {
 24 		this._tag = tag;
 25 		// call the super class
 26 		// 2nd parameter is passed to EventDispatcher constructor
 27 		Event.call(this, type);
 28 	};
 29 
 30 	/** @constant */
 31 	RFIDEvent.ADD_TAG = "addTag";
 32 	/** @constant */
 33 	RFIDEvent.REMOVE_TAG = "removeTag";
 34 
 35 	RFIDEvent.prototype = JSUTILS.inherit(Event.prototype);
 36 	RFIDEvent.prototype.constructor = RFIDEvent;
 37 
 38 	/**
 39 	 * [read-only] The RFID tag value (hexadecimal string).
 40 	 * @name RFIDEvent#tag
 41 	 * @property
 42 	 * @type String
 43 	 */ 
 44 	RFIDEvent.prototype.__defineGetter__("tag", function() { return this._tag; });
 45 
 46 	return RFIDEvent;
 47 
 48 }());

 51 JSUTILS.namespace('BO.custom.ID12RFIDReader');
 52 
 53 BO.custom.ID12RFIDReader = (function() {
 54 
 55 	var ID12RFIDReader;
 56 
 57 	// dependencies
 58 	var EventDispatcher = JSUTILS.EventDispatcher,
 59 		IOBoardEvent = BO.IOBoardEvent,
 60 		RFIDEvent = BO.io.RFIDEvent;
 61 
 62 	/**
 63 	 * Innovations ID-12 RFID Reader.
 64 	 * <p>To use this object, RFIDFirmata must be uploaded to the
 65 	 * IOBoard rather than StandardFirmata. See custom_examples/readme.txt
 66 	 * for insturctions.</p>
 67 	 *
 68 	 * @exports ID12RFIDReader as BO.custom.ID12RFIDReader
 69 	 * @constructor
 70 	 * @param {IOBoard} board A reference to the IOBoard instance
 71 	 */
 72 	ID12RFIDReader = function(board) {
 73 		"use strict";
 74 
 75 		this.name = "ID12RFIDReader"; 	// for testing
 76 
 77 		// to do: object id to be passed in as a param rather than
 78 		// explicitly set? This would support the ability to use multiple
 79 		// RFID readers with a single Arduino.
 80 		// Would also need to update the Arduino library to enable this.
 81 		this.ID12_READER = 13,
 82 		this.READ_EVENT = 1,
 83 		this.REMOVE_EVENT = 2;
 84 
 85 		this._board = board;
 86 		
 87 		this._evtDispatcher = new EventDispatcher(this);
 88 
 89 		board.addEventListener(IOBoardEvent.SYSEX_MESSAGE, this.onSysExMessage.bind(this));			
 90 	};
 91 
 92 	ID12RFIDReader.prototype = {
 93 
 94 		// private methods:
 95 		/**
 96 		 * @private
 97 		 */
 98 		onSysExMessage: function(event) {
 99 			var message = event.message;
100 
101 			if (message[0] != this.ID12_READER) {
102 				return;
103 			} else {
104 				this.processRFIDData(message);
105 			}
106 		},
107 
108 		// this is nice! found it here:
109 		// http://stackoverflow.com/questions/57803/how-to-convert-decimal-to-hex-in-javascript
110 		/**
111 		 * @private
112 		 */
113 		dec2hex: function(i) {
114 	   		return (i+0x100).toString(16).substr(-2).toUpperCase();
115 		},
116 		
117 		/**
118 		 * @private
119 		 */
120 		processRFIDData: function(data) {
121 
122 			var tagEvent = this._board.getValueFromTwo7bitBytes(data[1], data[2]);
123 			var tagEventType = "";
124 			var tag = "";
125 					
126 			for (var i=3, len=data.length; i<len; i+=2) {
127 				tag += this.dec2hex(this._board.getValueFromTwo7bitBytes(data[i], data[i+1]));
128 			}
129 
130 			// change this to dispatch a single event and handle add or remove in object parameter?
131 			if (tagEvent == this.READ_EVENT) {
132 				this.dispatch(new RFIDEvent(RFIDEvent.ADD_TAG, tag));
133 			} else if (tagEvent == this.REMOVE_EVENT) {
134 				this.dispatch(new RFIDEvent(RFIDEvent.REMOVE_TAG, tag));
135 			} else {
136 				// got something else
137 				return;
138 			}
139 			
140 		},
141 		
142 		/**
143 		 * @private
144 		 */
145 		dispatch: function(event) {
146 			this.dispatchEvent(event);
147 		},
148 		
149 		// public methods:
150 		
151 		/* implement EventDispatcher */
152 		
153 		/**
154 		 * @param {String} type The event type
155 		 * @param {Function} listener The function to be called when the event is fired
156 		 */
157 		addEventListener: function(type, listener) {
158 			this._evtDispatcher.addEventListener(type, listener);
159 		},
160 		
161 		/**
162 		 * @param {String} type The event type
163 		 * @param {Function} listener The function to be called when the event is fired
164 		 */
165 		removeEventListener: function(type, listener) {
166 			this._evtDispatcher.removeEventListener(type, listener);
167 		},
168 		
169 		/**
170 		 * @param {String} type The event type
171 		 * return {boolean} True is listener exists for this type, false if not.
172 		 */
173 		hasEventListener: function(type) {
174 			return this._evtDispatcher.hasEventListener(type);
175 		},
176 		
177 		/**
178 		 * @param {Event} type The Event object
179 		 * @param {Object} optionalParams Optional parameters to assign to the event object.
180 		 * return {boolean} True if dispatch is successful, false if not.
181 		 */		
182 		dispatchEvent: function(event, optionalParams) {
183 			return this._evtDispatcher.dispatchEvent(event, optionalParams);
184 		}		
185 
186 	};
187 
188 
189 	// document events
190 
191 	/**
192 	 * The addTag event is dispatched when a new tag is read.
193 	 * @name ID12RFIDReader#addTag
194 	 * @type BO.custom.RFIDEvent.ADD_TAG
195 	 * @event
196 	 * @param {BO.custom.ID12RFIDReader} target A reference to the ID12RFIDReader object.
197 	 * @param {String} tag The RFID tag value.	 
198 	 */
199 	 
200 	/**
201 	 * The removeTag event is dispatched when a tag is removed from the reader.
202 	 * @name ID12RFIDReader#removeTag
203 	 * @type BO.custom.RFIDEvent.REMOVE_TAG
204 	 * @event
205 	 * @param {BO.custom.ID12RFIDReader} target A reference to the ID12RFIDReader object.
206 	 * @param {String} tag The RFID tag value.	 
207 	 */		 	
208 
209 	return ID12RFIDReader;
210 
211 }());
