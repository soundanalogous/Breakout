  6 JSUTILS.namespace('BO.custom.ID12RFIDReader');
  7 
  8 BO.custom.ID12RFIDReader = (function() {
  9 
 10 	var ID12RFIDReader;
 11 
 12 	// private static constants:
 13 	var	ID12_READER = 13,
 14 		READ_EVENT = 1,
 15 		REMOVE_EVENT = 2;
 16 
 17 	// dependencies
 18 	var EventDispatcher = JSUTILS.EventDispatcher,
 19 		IOBoardEvent = BO.IOBoardEvent,
 20 		RFIDEvent = BO.custom.RFIDEvent;
 21 
 22 	/**
 23 	 * Innovations ID-12 RFID Reader.
 24 	 * <p>To use this object with standard io objects in Breakout, 
 25 	 * RFIDFirmata must be uploaded to the IOBoard rather than StandardFirmata. 
 26 	 * See custom_examples/readme.txt for insturctions.</p>
 27 	 *
 28 	 * <p>Is is also possible to create a custom application for your
 29 	 * IOBoard that includes the RFID reader. See IDx_Reader_Firmata_Example
 30 	 * in the IDxRFIDReader library example files.</p>
 31 	 *
 32 	 * @exports ID12RFIDReader as BO.custom.ID12RFIDReader
 33 	 * @class Creates an interface to an ID-12 RFID Reader. Other Innovations
 34 	 * RFID readers will likely work but have not been tested. This object
 35 	 * requires firmware other than StandardFirmata to be uploaded to the I/O board.
 36 	 * See Breakout/custom_examples/rfid_example1.html and rfid_example2.html for
 37 	 * example applications.
 38 	 * @constructor
 39 	 * @param {IOBoard} board A reference to the IOBoard instance
 40 	 * @param {Number} readerId The ID assigned to the reader in the firmware
 41 	 * running on the IOBoard (default = 13)
 42 	 */
 43 	ID12RFIDReader = function(board, readerId) {
 44 		"use strict";
 45 
 46 		this.name = "ID12RFIDReader";
 47 
 48 		this._readerId = readerId || ID12_READER;
 49 		this._board = board;
 50 		this._evtDispatcher = new EventDispatcher(this);
 51 
 52 		board.addEventListener(IOBoardEvent.SYSEX_MESSAGE, this.onSysExMessage.bind(this));			
 53 	};
 54 
 55 	ID12RFIDReader.prototype = {
 56 
 57 		// private methods:
 58 		/**
 59 		 * @private
 60 		 */
 61 		onSysExMessage: function(event) {
 62 			var message = event.message;
 63 
 64 			if (message[0] != this._readerId) {
 65 				return;
 66 			} else {
 67 				this.processRFIDData(message);
 68 			}
 69 		},
 70 
 71 		// this is nice! found it here:
 72 		// http://stackoverflow.com/questions/57803/how-to-convert-decimal-to-hex-in-javascript
 73 		/**
 74 		 * @private
 75 		 */
 76 		dec2hex: function(i) {
 77 	   		return (i+0x100).toString(16).substr(-2).toUpperCase();
 78 		},
 79 		
 80 		/**
 81 		 * @private
 82 		 */
 83 		processRFIDData: function(data) {
 84 
 85 			var tagEvent = this._board.getValueFromTwo7bitBytes(data[1], data[2]);
 86 			var tagEventType = "";
 87 			var tag = "";
 88 					
 89 			for (var i=3, len=data.length; i<len; i+=2) {
 90 				tag += this.dec2hex(this._board.getValueFromTwo7bitBytes(data[i], data[i+1]));
 91 			}
 92 
 93 			// change this to dispatch a single event and handle add or remove in object parameter?
 94 			if (tagEvent == READ_EVENT) {
 95 				this.dispatch(new RFIDEvent(RFIDEvent.ADD_TAG, tag));
 96 			} else if (tagEvent == REMOVE_EVENT) {
 97 				this.dispatch(new RFIDEvent(RFIDEvent.REMOVE_TAG, tag));
 98 			} else {
 99 				// got something else
100 				return;
101 			}
102 			
103 		},
104 		
105 		/**
106 		 * @private
107 		 */
108 		dispatch: function(event) {
109 			this.dispatchEvent(event);
110 		},
111 		
112 		// public methods:
113 		
114 		/* implement EventDispatcher */
115 		
116 		/**
117 		 * @param {String} type The event type
118 		 * @param {Function} listener The function to be called when the event is fired
119 		 */
120 		addEventListener: function(type, listener) {
121 			this._evtDispatcher.addEventListener(type, listener);
122 		},
123 		
124 		/**
125 		 * @param {String} type The event type
126 		 * @param {Function} listener The function to be called when the event is fired
127 		 */
128 		removeEventListener: function(type, listener) {
129 			this._evtDispatcher.removeEventListener(type, listener);
130 		},
131 		
132 		/**
133 		 * @param {String} type The event type
134 		 * return {boolean} True is listener exists for this type, false if not.
135 		 */
136 		hasEventListener: function(type) {
137 			return this._evtDispatcher.hasEventListener(type);
138 		},
139 		
140 		/**
141 		 * @param {Event} type The Event object
142 		 * @param {Object} optionalParams Optional parameters to assign to the event object.
143 		 * return {boolean} True if dispatch is successful, false if not.
144 		 */		
145 		dispatchEvent: function(event, optionalParams) {
146 			return this._evtDispatcher.dispatchEvent(event, optionalParams);
147 		}		
148 
149 	};
150 
151 
152 	// document events
153 
154 	/**
155 	 * The addTag event is dispatched when a new tag is read.
156 	 * @name ID12RFIDReader#addTag
157 	 * @type BO.custom.RFIDEvent.ADD_TAG
158 	 * @event
159 	 * @param {BO.custom.ID12RFIDReader} target A reference to the ID12RFIDReader object.
160 	 * @param {String} tag The RFID tag value.	 
161 	 */
162 	 
163 	/**
164 	 * The removeTag event is dispatched when a tag is removed from the reader.
165 	 * @name ID12RFIDReader#removeTag
166 	 * @type BO.custom.RFIDEvent.REMOVE_TAG
167 	 * @event
168 	 * @param {BO.custom.ID12RFIDReader} target A reference to the ID12RFIDReader object.
169 	 * @param {String} tag The RFID tag value.	 
170 	 */		 	
171 
172 	return ID12RFIDReader;
173 
174 }());
