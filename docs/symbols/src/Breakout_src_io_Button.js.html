 50 JSUTILS.namespace('BO.io.Button');
 51 
 52 BO.io.Button = (function() {
 53 
 54 	var Button;
 55 
 56 	// dependencies
 57 	var PhysicalInputBase = BO.PhysicalInputBase,
 58 		Event = JSUTILS.Event,
 59 		Pin = BO.Pin,
 60 		ButtonEvent = BO.io.ButtonEvent;
 61 
 62 	/**
 63 	 * An object to represent a physical button. The advantage of using the Button class over
 64 	 * listening for pin change events on a Pin object, is that the Button class handles debouncing 
 65 	 * and provides helpful button events: Pressed, Released, Long Press and Sustained Press
 66 	 *
 67 	 * <p>PULL_UP vs PULL_DOWN. If the other end of the resistor connected to the button is 
 68 	 * connected to ground, configuration is PULL_DOWN, if the resistor is connected to power, 
 69 	 * then the configuration is PULL_UP.</p>
 70 	 *
 71 	 * @exports Button as BO.io.Button
 72 	 * @constructor
 73 	 * @augments BO.PhysicalInputBase
 74 	 * @param {IOBoard} board A reference to the IOBoard instance
 75 	 * @param {Pin} pin A reference to the Pin the button is connected to.
 76 	 * @param {Number} buttonMode The mode of the button (either Button.PULL_DOWN or 
 77 	 * Button.PULL_UP if wired with external resistors or Button.INTERNAL_PULL_UP if
 78 	 * using the internal pull-up resistors. Default is PULL_DOWN.
 79 	 * @param {Number} sustainedPressInterval The delay time in milliseconds before a sustained press event is fired.
 80 	 *
 81 	 */
 82 	Button = function(board, pin, buttonMode, sustainedPressInterval) {
 83 		"use strict";
 84 		
 85 		PhysicalInputBase.call(this);
 86 
 87 		this.name = "Button";
 88 		this._pin = pin;
 89 
 90 		var pinNumber = pin.number;
 91 		
 92 		this.buttonMode = buttonMode || Button.PULL_DOWN;
 93 		this._sustainedPressInterval = sustainedPressInterval || 1000;
 94 
 95 		this._debounceInterval = 20,
 96 		this._repeatCount = 0,
 97 		this._timer = null,
 98 		this._timeout = null;
 99 		
100 		this._board = board;
101 		board.setDigitalPinMode(pinNumber, Pin.DIN);
102 
103 		if (this.buttonMode === Button.INTERNAL_PULL_UP) {
104 			// enable internal pull up resistor
105 			board.enablePullUp(pinNumber);
106 			// set value to high to avoid initial change event
107 			this._pin.value = Pin.HIGH;
108 		}
109 				
110 		this._pin.addEventListener(Event.CHANGE, this.onPinChange.bind(this));	
111 	};
112 
113 
114 	Button.prototype = JSUTILS.inherit(PhysicalInputBase.prototype);
115 	Button.prototype.constructor = Button;
116 
117 	/**
118 	 * @private
119 	 */
120 	Button.prototype.onPinChange = function(evt) {
121 		
122 		var btnVal = evt.target.value;
123 		var stateHandler;
124 				
125 		if (this.buttonMode === Button.PULL_DOWN) {
126 			if (btnVal === 1) stateHandler = this.pressed;
127 			else stateHandler = this.released;
128 		} else if (this.buttonMode === Button.PULL_UP || this.buttonMode === Button.INTERNAL_PULL_UP) {
129 			if (btnVal === 1) stateHandler = this.released;
130 			else stateHandler = this.pressed;
131 		}
132 		
133 		if (this._timeout === null) {
134 			this._timeout = setTimeout(stateHandler.bind(this), this._debounceInterval);
135 		} else {
136 			clearTimeout(this._timeout);
137 			this._timeout = setTimeout(stateHandler.bind(this), this._debounceInterval);
138 		}
139 	};
140 	
141 	/**
142 	 * @private
143 	 */
144 	Button.prototype.pressed = function() {
145 		this._timeout = null;
146 
147 		this.dispatchEvent(new ButtonEvent(ButtonEvent.PRESS));
148 		
149 		this._timer = setInterval(this.sustainedPress.bind(this), this._sustainedPressInterval);
150 	};
151 	
152 	/**
153 	 * @private
154 	 */	
155 	Button.prototype.released = function() {
156 		this._timeout = null;
157 		this.dispatchEvent(new ButtonEvent(ButtonEvent.RELEASE));
158 		
159 		if (this._timer != null) {
160 			clearInterval(this._timer);
161 			this._timer = null;
162 		}
163 		
164 		this._repeatCount = 0;
165 	};
166 	
167 	/**
168 	 * @private
169 	 */
170 	Button.prototype.sustainedPress = function() {
171 		if (this._repeatCount > 0) {
172 			this.dispatchEvent(new ButtonEvent(ButtonEvent.SUSTAINED_PRESS));
173 		} else {
174 			this.dispatchEvent(new ButtonEvent(ButtonEvent.LONG_PRESS));
175 		}
176 		
177 		this._repeatCount++;
178 	};
179 	
180 	/**
181 	 * The debounce time interval in milliseconds.
182 	 * @name Button#debounceInterval
183 	 * @property
184 	 * @type Number
185 	 */ 
186 	Button.prototype.__defineGetter__("debounceInterval", function() { return this._debounceInterval; });
187 	Button.prototype.__defineSetter__("debounceInterval", function(interval) { this._debounceInterval = interval; });
188 	
189 	/**
190 	 * The delay time (in milliseconds) the button must be held before a sustained press event is fired.
191 	 * @name Button#sustainedPressInterval
192 	 * @property
193 	 * @type Number
194 	 */ 
195 	Button.prototype.__defineGetter__("sustainedPressInterval", function() { return this._sustainedPressInterval; });
196 	Button.prototype.__defineSetter__("sustainedPressInterval", function(intervalTime) { this._sustainedPressInterval = intervalTime; });
197 
198 	/**
199 	 * [read-only] The pin number of the pin the button is attached to.
200 	 * @name Button#pinNumber
201 	 * @property
202 	 * @type Number
203 	 */
204 	Button.prototype.__defineGetter__("pinNumber", function() { return this._pin.number; });	
205 
206 	/** @constant */
207 	Button.PULL_DOWN = 0;
208 	/** @constant */
209 	Button.PULL_UP 	= 1;
210 	/** @contstant */
211 	Button.INTERNAL_PULL_UP = 2;
212 
213 
214 	// document events
215 
216 	/**
217 	 * The pressed event is dispatched when the button is pressed.
218 	 * @name Button#pressed
219 	 * @type BO.io.ButtonEvent.PRESS
220 	 * @event
221 	 * @param {BO.io.Button} target A reference to the Button object
222 	 */	
223 
224 	/**
225 	 * The released event is dispatched when the button is released.
226 	 * @name Button#released
227 	 * @type BO.io.ButtonEvent.RELEASE
228 	 * @event
229 	 * @param {BO.io.Button} target A reference to the Button object
230 	 */	
231 	 
232 	/**
233 	 * The longPress event is dispatched once when the button has been held for
234 	 * the time duration specified by the sustainedPressInterval property.
235 	 * @name Button#longPress
236 	 * @type BO.io.ButtonEvent.LONG_PRESS
237 	 * @event
238 	 * @param {BO.io.Button} target A reference to the Button object
239 	 */	
240 	 
241 	/**
242 	 * The sustainedPress event is dispatched continuously at the rate 
243 	 * specified by the sustainedPressInterval property while the button is held.
244 	 * @name Button#sustainedPress
245 	 * @type BO.io.ButtonEvent.SUSTAINED_PRESS
246 	 * @event
247 	 * @param {BO.io.Button} target A reference to the Button object
248 	 */		 	 	 
249 
250 	return Button;
251 
252 }());
