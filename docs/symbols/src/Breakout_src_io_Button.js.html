  6 JSUTILS.namespace('BO.io.Button');
  7 
  8 BO.io.Button = (function() {
  9 
 10 	var Button;
 11 
 12 	// dependencies
 13 	var PhysicalInputBase = BO.PhysicalInputBase,
 14 		PinEvent = BO.PinEvent,
 15 		Pin = BO.Pin,
 16 		ButtonEvent = BO.io.ButtonEvent;
 17 
 18 	/**
 19 	 * An object to represent a physical button. The advantage of using the 
 20 	 * Button class over listening for pin change events on a Pin object, is 
 21 	 * that the Button class handles debouncing and provides helpful button
 22 	 * events: Pressed, Released, Long Press and Sustained Press
 23 	 *
 24 	 * <p>PULL_UP vs PULL_DOWN. If the other end of the resistor connected to
 25 	 * the button is connected to ground, configuration is PULL_DOWN, if the 
 26 	 * resistor is connected to power, then the configuration is PULL_UP.</p>
 27 	 *
 28 	 * @exports Button as BO.io.Button
 29 	 * @class Creates and interface to a physical button. The Button object
 30 	 * dispatches events on state changes such as Pressed, Released and 
 31 	 * Sustained Press. The Button object also handles debouncing.
 32 	 * @constructor
 33 	 * @augments BO.PhysicalInputBase
 34 	 * @param {IOBoard} board A reference to the IOBoard instance
 35 	 * @param {Pin} pin A reference to the Pin the button is connected to.
 36 	 * @param {Number} buttonMode The mode of the button (either 
 37 	 * Button.PULL_DOWN or Button.PULL_UP if wired with external resistors or 
 38 	 * Button.INTERNAL_PULL_UP if using the internal pull-up resistors. Default
 39 	 * is PULL_DOWN.
 40 	 * @param {Number} sustainedPressInterval The delay time in milliseconds 
 41 	 * before a sustained press event is fired.
 42 	 */
 43 	Button = function(board, pin, buttonMode, sustainedPressInterval) {
 44 		"use strict";
 45 		
 46 		PhysicalInputBase.call(this);
 47 
 48 		this.name = "Button";
 49 		this._pin = pin;
 50 
 51 		var pinNumber = pin.number;
 52 		
 53 		this.buttonMode = buttonMode || Button.PULL_DOWN;
 54 		this._sustainedPressInterval = sustainedPressInterval || 1000;
 55 
 56 		this._debounceInterval = 20,
 57 		this._repeatCount = 0,
 58 		this._timer = null,
 59 		this._timeout = null;
 60 		
 61 		this._board = board;
 62 		board.setDigitalPinMode(pinNumber, Pin.DIN);
 63 
 64 		if (this.buttonMode === Button.INTERNAL_PULL_UP) {
 65 			// Enable internal pull up resistor
 66 			board.enablePullUp(pinNumber);
 67 			// Set value to high to avoid initial change event
 68 			this._pin.value = Pin.HIGH;
 69 		} else if (this.buttonMode === Button.PULL_UP) {
 70 			// Set value to high to avoid initial change event
 71 			this._pin.value = Pin.HIGH;
 72 		}
 73 		this._pin.addEventListener(PinEvent.CHANGE, this.onPinChange.bind(this));	
 74 	};
 75 
 76 	Button.prototype = JSUTILS.inherit(PhysicalInputBase.prototype);
 77 	Button.prototype.constructor = Button;
 78 
 79 	/**
 80 	 * @private
 81 	 */
 82 	Button.prototype.onPinChange = function(evt) {
 83 		
 84 		var btnVal = evt.target.value;
 85 		var stateHandler;
 86 				
 87 		if (this.buttonMode === Button.PULL_DOWN) {
 88 			if (btnVal === 1) stateHandler = this.pressed;
 89 			else stateHandler = this.released;
 90 		} else if (this.buttonMode === Button.PULL_UP || this.buttonMode === Button.INTERNAL_PULL_UP) {
 91 			if (btnVal === 1) stateHandler = this.released;
 92 			else stateHandler = this.pressed;
 93 		}
 94 		
 95 		if (this._timeout === null) {
 96 			this._timeout = setTimeout(stateHandler.bind(this), this._debounceInterval);
 97 		} else {
 98 			clearTimeout(this._timeout);
 99 			this._timeout = setTimeout(stateHandler.bind(this), this._debounceInterval);
100 		}
101 	};
102 	
103 	/**
104 	 * @private
105 	 */
106 	Button.prototype.pressed = function() {
107 		this._timeout = null;
108 
109 		this.dispatchEvent(new ButtonEvent(ButtonEvent.PRESS));
110 		
111 		this._timer = setInterval(this.sustainedPress.bind(this), this._sustainedPressInterval);
112 	};
113 	
114 	/**
115 	 * @private
116 	 */	
117 	Button.prototype.released = function() {
118 		this._timeout = null;
119 		this.dispatchEvent(new ButtonEvent(ButtonEvent.RELEASE));
120 		
121 		if (this._timer != null) {
122 			clearInterval(this._timer);
123 			this._timer = null;
124 		}
125 		
126 		this._repeatCount = 0;
127 	};
128 	
129 	/**
130 	 * @private
131 	 */
132 	Button.prototype.sustainedPress = function() {
133 		if (this._repeatCount > 0) {
134 			this.dispatchEvent(new ButtonEvent(ButtonEvent.SUSTAINED_PRESS));
135 		} else {
136 			this.dispatchEvent(new ButtonEvent(ButtonEvent.LONG_PRESS));
137 		}
138 		
139 		this._repeatCount++;
140 	};
141 	
142 	/**
143 	 * The debounce time interval in milliseconds.
144 	 * @name Button#debounceInterval
145 	 * @property
146 	 * @type Number
147 	 */ 
148 	Button.prototype.__defineGetter__("debounceInterval", function() { return this._debounceInterval; });
149 	Button.prototype.__defineSetter__("debounceInterval", function(interval) { this._debounceInterval = interval; });
150 	
151 	/**
152 	 * The delay time (in milliseconds) the button must be held before a
153 	 * sustained press event is fired.
154 	 * 
155 	 * @name Button#sustainedPressInterval
156 	 * @property
157 	 * @type Number
158 	 */ 
159 	Button.prototype.__defineGetter__("sustainedPressInterval", function() { return this._sustainedPressInterval; });
160 	Button.prototype.__defineSetter__("sustainedPressInterval", function(intervalTime) { this._sustainedPressInterval = intervalTime; });
161 
162 	/**
163 	 * [read-only] The pin number of the pin the button is attached to.
164 	 * 
165 	 * @name Button#pinNumber
166 	 * @property
167 	 * @type Number
168 	 */
169 	Button.prototype.__defineGetter__("pinNumber", function() { return this._pin.number; });	
170 
171 	/** @constant */
172 	Button.PULL_DOWN = 0;
173 	/** @constant */
174 	Button.PULL_UP 	= 1;
175 	/** @contstant */
176 	Button.INTERNAL_PULL_UP = 2;
177 
178 
179 	// Document events
180 
181 	/**
182 	 * The pressed event is dispatched when the button is pressed.
183 	 * @name Button#pressed
184 	 * @type BO.io.ButtonEvent.PRESS
185 	 * @event
186 	 * @param {BO.io.Button} target A reference to the Button object
187 	 */	
188 
189 	/**
190 	 * The released event is dispatched when the button is released.
191 	 * @name Button#released
192 	 * @type BO.io.ButtonEvent.RELEASE
193 	 * @event
194 	 * @param {BO.io.Button} target A reference to the Button object
195 	 */	
196 	 
197 	/**
198 	 * The longPress event is dispatched once when the button has been held for
199 	 * the time duration specified by the sustainedPressInterval property.
200 	 * @name Button#longPress
201 	 * @type BO.io.ButtonEvent.LONG_PRESS
202 	 * @event
203 	 * @param {BO.io.Button} target A reference to the Button object
204 	 */	
205 	 
206 	/**
207 	 * The sustainedPress event is dispatched continuously at the rate 
208 	 * specified by the sustainedPressInterval property while the button is
209 	 * held.
210 	 * @name Button#sustainedPress
211 	 * @type BO.io.ButtonEvent.SUSTAINED_PRESS
212 	 * @event
213 	 * @param {BO.io.Button} target A reference to the Button object
214 	 */		 	 	 
215 
216 	return Button;
217 
218 }());
